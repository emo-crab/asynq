1//! Redis 配置和连接管理模块
2//! Redis configuration and connection management module
3
4use crate::error::Result;
5use async_trait::async_trait;
6use redis::{Cmd, ConnectionInfo, IntoConnectionInfo, Pipeline, RedisFuture, Value};
7
8pub enum RedisConnection {
9  Single(redis::aio::MultiplexedConnection),
10  #[cfg(feature = "cluster")]
11  Cluster(redis::cluster_async::ClusterConnection),
12}
13#[async_trait]
14impl redis::aio::ConnectionLike for RedisConnection {
15  fn req_packed_command<'a>(&'a mut self, cmd: &'a Cmd) -> RedisFuture<'a, Value> {
16    match self {
17      RedisConnection::Single(c) => c.req_packed_command(cmd),
18      #[cfg(feature = "cluster")]
19      RedisConnection::Cluster(c) => c.req_packed_command(cmd),
20    }
21  }
22
23  fn req_packed_commands<'a>(
24    &'a mut self,
25    cmd: &'a Pipeline,
26    offset: usize,
27    count: usize,
28  ) -> RedisFuture<'a, Vec<Value>> {
29    match self {
30      RedisConnection::Single(c) => c.req_packed_commands(cmd, offset, count),
31      #[cfg(feature = "cluster")]
32      RedisConnection::Cluster(c) => c.req_packed_commands(cmd, offset, count),
33    }
34  }
35
36  /// Gets the current Redis database number
37  /// # Notes
38  /// * Always returns 0 as database selection is not supported in cluster mode
39  /// # Returns
40  /// * `i64` - The database number (always 0)
41  fn get_db(&self) -> i64 {
42    0
43  }
44}

46/// Redis 连接类型
47/// Redis connection types
48#[derive(Debug, Clone)]
49pub enum RedisConnectionType {
50  /// 单机连接
51  /// Standalone connection
52  Single(ConnectionInfo),
53  #[cfg(feature = "cluster")]
54  /// 集群连接
55  /// Cluster connection
56  Cluster(Vec<ConnectionInfo>),
57}
58
59impl RedisConnectionType {
60  /// 创建单机连接
61  /// Create a standalone connection
62  pub fn single<T: IntoConnectionInfo>(connection_info: T) -> Result<Self> {
63    Ok(Self::Single(connection_info.into_connection_info()?))
64  }
65  #[cfg(feature = "cluster")]
66  /// 创建集群连接
67  /// Create a cluster connection
68  pub fn cluster<T: IntoConnectionInfo>(nodes: Vec<T>) -> Result<Self> {
69    let nodes = nodes
70      .into_iter()
71      .map(|x| x.into_connection_info())
72      .filter_map(|x| x.ok())
73      .collect::<Vec<_>>();
74    if nodes.is_empty() {
75      return Err(crate::error::Error::other(
76        "At least one valid node is required for cluster connection",
77      ));
78    }
79    Ok(Self::Cluster(nodes))
80  }
