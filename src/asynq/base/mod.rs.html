186/// 服务器状态
187/// Server state
188#[derive(Debug, Clone, PartialEq)]
189pub enum ServerState {
190  /// 新建状态
191  /// New
192  New,
193  /// 活跃状态
194  /// Active
195  Active,
196  /// 已停止状态
197  /// Stopped
198  Stopped,
199  /// 已关闭状态
200  /// Closed
201  Closed,
202}
203
204impl ServerState {
205  /// 转换为字符串
206  /// Convert to string
207  pub fn as_str(&self) -> &'static str {
208    match self {
209      Self::New => "new",
210      Self::Active => "active",
211      Self::Stopped => "stopped",
212      Self::Closed => "closed",
213    }
214  }
215}
216
217impl FromStr for ServerState {
218  type Err = ();
219
220  fn from_str(s: &str) -> std::result::Result<Self, Self::Err> {
221    match s {
222      "new" => Ok(Self::New),
223      "active" => Ok(Self::Active),
224      "stopped" => Ok(Self::Stopped),
225      "closed" => Ok(Self::Closed),
226      _ => Err(()),
227    }
228  }
229}
