31/// 任务状态
32/// Task state
33#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
34pub enum TaskState {
35  /// 任务正在被处理
36  /// Task is being processed
37  Active,
38  /// 任务准备好被处理
39  /// Task is ready to be processed
40  Pending,
41  /// 任务被安排在将来某个时间处理
42  /// Task is scheduled to be processed at a later time
43  Scheduled,
44  /// 任务之前失败了，安排在将来某个时间重试
45  /// Task has failed before, scheduled to retry at a later time
46  Retry,
47  /// 任务被归档并存储以供检查
48  /// Task is archived and stored for inspection
49  Archived,
50  /// 任务处理成功并保留到保留 TTL 过期
51  /// Task is successfully processed and retained until retention TTL expires
52  Completed,
53  /// 任务在组中等待聚合
54  /// Task is waiting for aggregation in the group
55  Aggregating,
56}
57
58impl TaskState {
59  /// 将任务状态转换为字符串
60  /// Convert task state to string
61  pub fn as_str(&self) -> &'static str {
62    match self {
63      Self::Active => "active",
64      Self::Pending => "pending",
65      Self::Scheduled => "scheduled",
66      Self::Retry => "retry",
67      Self::Archived => "archived",
68      Self::Completed => "completed",
69      Self::Aggregating => "aggregating",
70    }
71  }
72  pub fn queue_key(&self, qname: &str, gname: Option<&str>) -> String {
73    match self {
74      Self::Active => active_key(qname),
75      Self::Pending => pending_key(qname),
76      Self::Scheduled => scheduled_key(qname),
77      Self::Retry => retry_key(qname),
78      Self::Archived => archived_key(qname),
79      Self::Completed => completed_key(qname),
80      Self::Aggregating => aggregating_key(qname, gname.unwrap_or("")), // 需要提供组名
81    }
82  }
83}

85impl FromStr for TaskState {
86  type Err = ();
87
88  fn from_str(s: &str) -> Result<Self, Self::Err> {
89    match s {
90      "active" => Ok(Self::Active),
91      "pending" => Ok(Self::Pending),
92      "scheduled" => Ok(Self::Scheduled),
93      "retry" => Ok(Self::Retry),
94      "archived" => Ok(Self::Archived),
95      "completed" => Ok(Self::Completed),
96      "aggregating" => Ok(Self::Aggregating),
97      _ => Err(()),
98    }
99  }
100}

180/// 生成唯一键 - 对应 Go 的 UniqueKey，使用 MD5 校验和
181/// Generate unique key - Corresponds to Go's UniqueKey, using MD5 checksum
182pub fn unique_key(qname: &str, task_type: &str, payload: &[u8]) -> String {
183  if payload.is_empty() {
184    return format!("{}unique:{}:", queue_key_prefix(qname), task_type);
185  }
186
187  // 使用 MD5 哈希与 Go 版本保持兼容
188  // Use MD5 hash to保持兼容 with Go version
189  let digest = md5::compute(payload);
190  let checksum = format!("{digest:x}");
191
192  format!(
193    "{}unique:{}:{}",
194    queue_key_prefix(qname),
195    task_type,
196    checksum
197  )
198}

312  use crate::base::keys;

314  #[test]
315  fn test_keys_generation() {
316    // 测试与 Go 版本兼容的键生成
317    // Test key generation compatible with Go version
318    assert_eq!(keys::pending_key("default"), "asynq:{default}:pending");
319    assert_eq!(keys::pending_key("default"), "asynq:{default}:pending"); // 别名
320    assert_eq!(keys::active_key("default"), "asynq:{default}:active");
321    assert_eq!(keys::scheduled_key("default"), "asynq:{default}:scheduled");
322    assert_eq!(keys::retry_key("default"), "asynq:{default}:retry");
323    assert_eq!(keys::archived_key("default"), "asynq:{default}:archived");
324    assert_eq!(keys::completed_key("default"), "asynq:{default}:completed");
325
326    // 测试全局键
327    // Test global keys
328    assert_eq!(keys::ALL_SERVERS, "asynq:servers");
329    assert_eq!(keys::ALL_WORKERS, "asynq:workers");
330    assert_eq!(keys::ALL_QUEUES, "asynq:queues");
331
332    // 测试服务器和工作者键（新格式）
333    // Test server and worker keys (new format)
334    assert_eq!(
335      keys::server_info_key("localhost", 12345, "server1"),
336      "asynq:servers:{localhost:12345:server1}"
337    );
338    assert_eq!(
339      keys::workers_key("localhost", 12345, "server1"),
340      "asynq:workers:{localhost:12345:server1}"
341    );
342
343    // 测试任务和组相关键
344    // Test task and group related keys
345    assert_eq!(
346      keys::task_key("default", "task1"),
347      "asynq:{default}:t:task1"
348    );
349    assert_eq!(
350      keys::group_key("default", "group1"),
351      "asynq:{default}:g:group1"
352    );
353
354    // 测试唯一键
355    // Test unique key
356    let unique_key = keys::unique_key("default", "email:send", b"test payload");
357    assert!(unique_key.starts_with("asynq:{default}:unique:email:send:"));
358
359    let empty_unique_key = keys::unique_key("default", "email:send", b"");
360    assert_eq!(empty_unique_key, "asynq:{default}:unique:email:send:");
361    assert_eq!(
362      keys::task_key("default", "task1"),
363      "asynq:{default}:t:task1"
364    );
365  }
