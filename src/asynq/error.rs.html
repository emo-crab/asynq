1//! 错误处理模块
2//! Error handling module

4//! 定义了 Asynq 库中使用的各种错误类型
5//! Defines various error types used in the Asynq library
6
7use thiserror::Error;
8
9/// Asynq 库的结果类型
10/// Result type for the Asynq library
11pub type Result<T> = std::result::Result<T, Error>;
12
13/// Asynq 错误类型
14/// Asynq error type
15#[derive(Error, Debug)]
16pub enum Error {
17  /// Redis 连接错误
18  /// Redis connection error
19  #[error("Redis connection error: {0}")]
20  Redis(#[from] redis::RedisError),
21  #[cfg(feature = "json")]
22  /// 序列化错误
23  /// Serialization error
24  #[error("Serialization error: {0}")]
25  Serialization(#[from] serde_json::Error),
26
27  /// Protocol Buffer 编码错误
28  /// Protocol buffer encoding error
29  #[error("Protocol buffer encoding error: {0}")]
30  ProtoEncode(#[from] prost::EncodeError),
31
32  /// Protocol Buffer 解码错误
33  /// Protocol buffer decoding error
34  #[error("Protocol buffer decoding error: {0}")]
35  ProtoDecode(#[from] prost::DecodeError),
36
37  /// 任务重复错误
38  /// Task already exists error
39  #[error("Task already exists")]
40  TaskDuplicate,
41
42  /// 任务 ID 冲突错误
43  /// Task ID conflict error
44  #[error("Task ID conflicts with another task")]
45  TaskIdConflict,
46
47  /// 任务未找到错误
48  /// Task not found error
49  #[error("Task not found: {id}")]
50  TaskNotFound { id: String },
51
52  /// 队列错误
53  /// Queue error
54  #[error("Queue error: {message}")]
55  Queue { message: String },
56
57  /// 无效的队列名称
58  /// Invalid queue name
59  #[error("Invalid queue name: {name}")]
60  InvalidQueueName { name: String },
61
62  /// 无效的任务类型
63  /// Invalid task type
64  #[error("Invalid task type: {task_type}")]
65  InvalidTaskType { task_type: String },
66
67  /// 服务器已关闭
68  /// Server closed
69  #[error("Server closed")]
70  ServerClosed,
71
72  /// 服务器已在运行
73  /// Server is already running
74  #[error("Server is already running")]
75  ServerRunning,
76
77  /// 超时错误
78  /// Timeout error
79  #[error("Operation timeout")]
80  Timeout,
81
82  /// 取消错误
83  /// Cancellation error
84  #[error("Operation cancelled")]
85  Cancelled,
86
87  /// 配置错误
88  /// Configuration error
89  #[error("Configuration error: {message}")]
90  Config { message: String },
91
92  /// IO 错误
93  /// IO error
94  #[error("IO error: {0}")]
95  Io(#[from] std::io::Error),
96
97  /// 其他错误
98  /// Other error
99  #[error("Other error: {message}")]
100  Other { message: String },
101
102  /// 未实现错误
103  /// Not implemented error
104  #[error("Not implemented: {0}")]
105  NotImplemented(String),
106}
107
108impl Error {
109  /// 创建队列错误
110  /// Create a queue error
111  pub fn queue<S: Into<String>>(message: S) -> Self {
112    Self::Queue {
113      message: message.into(),
114    }
115  }
116
117  /// 创建配置错误
118  /// Create a configuration error
119  pub fn config<S: Into<String>>(message: S) -> Self {
120    Self::Config {
121      message: message.into(),
122    }
123  }
124
125  /// 创建其他错误
126  /// Create another type of error
127  pub fn other<S: Into<String>>(message: S) -> Self {
128    Self::Other {
129      message: message.into(),
130    }
131  }
132
133  /// 检查是否为重试错误
134  /// Check if the error is retriable
135  pub fn is_retriable(&self) -> bool {
136    matches!(self, Self::Redis(_) | Self::Timeout | Self::Io(_))
137  }
138
139  /// 检查是否为致命错误
140  /// Check if the error is fatal
141  pub fn is_fatal(&self) -> bool {
142    !self.is_retriable()
143  }
144}

146/// 特殊的跳过重试错误包装器
147/// Special skip retry error wrapper
148#[derive(Error, Debug)]
149#[error("Skip retry: {0}")]
150pub struct SkipRetryError(pub Box<dyn std::error::Error + Send + Sync>);
151
152impl SkipRetryError {
153  /// 创建新的跳过重试错误
154  /// Create a new skip retry error
155  pub fn new<E>(error: E) -> Self
156  where
157    E: std::error::Error + Send + Sync + 'static,
158  {
159    Self(Box::new(error))
160  }
161}

171  /// Create a new revoke task error
172  pub fn new<E>(error: E) -> Self
173  where

175  {
176    Self(Box::new(error))
177  }
178}

180#[cfg(test)]
181mod tests {
182  use super::*;
183
184  #[test]
185  fn test_error_creation() {
186    let err = Error::queue("test queue error");
187    assert!(matches!(err, Error::Queue { .. }));
188
189    let err = Error::config("test config error");
190    assert!(matches!(err, Error::Config { .. }));
191
192    let err = Error::other("test other error");
193    assert!(matches!(err, Error::Other { .. }));
194  }
195
196  #[test]
197  fn test_error_retriable() {
198    assert!(Error::Timeout.is_retriable());
199    assert!(!Error::TaskDuplicate.is_retriable());
200    assert!(!Error::ServerClosed.is_retriable());
201  }
202
203  #[test]
204  fn test_skip_retry_error() {
205    let inner_err = std::io::Error::other("test error");
206    let skip_err = SkipRetryError::new(inner_err);
207    assert!(skip_err.to_string().contains("Skip retry"));
208  }
209
210  #[test]
211  fn test_revoke_task_error() {
212    let inner_err = std::io::Error::other("test error");
213    let revoke_err = RevokeTaskError::new(inner_err);
214    assert!(revoke_err.to_string().contains("Revoke task"));
215  }
