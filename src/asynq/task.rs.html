16/// 表示要执行的工作单元的任务
17/// Represents a task as a unit of work to be executed
18#[derive(Debug, Clone, PartialEq)]
19pub struct Task {
20  /// 任务类型名称
21  /// Task type name
22  pub task_type: String,
23  /// 任务负载数据
24  /// Task payload data
25  pub payload: Vec<u8>,
26  /// 任务选项
27  /// Task options
28  pub options: TaskOptions,
29}
30
31impl Task {
32  /// 创建新任务
33  /// Create a new task
34  pub fn new<T: AsRef<str>>(task_type: T, payload: &[u8]) -> Result<Self> {
35    let task_type = task_type.as_ref();
36    if task_type.trim().is_empty() {
37      return Err(Error::InvalidTaskType {
38        task_type: task_type.to_string(),
39      });
40    }
41
42    Ok(Self {
43      task_type: task_type.to_string(),
44      payload: payload.to_vec(),
45      options: TaskOptions::default(),
46    })
47  }
48  #[cfg(feature = "json")]
49  /// 使用 JSON 负载创建新任务
50  /// Create a new task with JSON payload
51  pub fn new_with_json<T: AsRef<str>, P: Serialize>(task_type: T, payload: &P) -> Result<Self> {
52    let json_payload = serde_json::to_vec(payload)?;
53    Self::new(task_type, &json_payload)
54  }
55
56  /// 设置任务选项
57  /// Set task options
58  pub fn with_options(mut self, options: TaskOptions) -> Self {
59    self.options = options;
60    self
61  }
62
63  /// 设置队列名称
64  /// Set queue name
65  pub fn with_queue<T: AsRef<str>>(mut self, queue: T) -> Self {
66    self.options.queue = queue.as_ref().to_string();
67    self
68  }
69
70  /// 设置最大重试次数
71  /// Set maximum retry attempts
72  pub fn with_max_retry(mut self, max_retry: i32) -> Self {
73    self.options.max_retry = max_retry.max(0);
74    self
75  }
76
77  /// 设置任务超时
78  /// Set task timeout
79  pub fn with_timeout(mut self, timeout: Duration) -> Self {
80    self.options.timeout = Some(timeout);
81    self
82  }
83

85  /// Set task deadline

87    self.options.deadline = Some(deadline);
88    self
89  }

92  /// Set unique task TTL

96  }

98  /// 设置任务组

105  /// 设置重试策略

117  }

120  /// Set task ID
121  pub fn with_task_id<T: AsRef<str>>(mut self, id: T) -> Self {
122    self.options.task_id = Some(id.as_ref().to_string());
123    self
124  }
125
126  /// 设置绝对处理时间
127  /// Set absolute processing time
128  pub fn with_process_at(mut self, when: DateTime<Utc>) -> Self {
129    self.options.process_at = Some(when);
130    self
131  }
132
133  /// 设置相对延迟
134  /// Set relative delay
135  pub fn with_process_in(mut self, delay: Duration) -> Self {
136    self.options.process_in = Some(delay);
137    self
138  }
139
140  /// 设置完成结果保留时间
141  /// Set retention time for completion results
142  pub fn with_retention(mut self, retention: Duration) -> Self {
143    self.options.retention = Some(retention);
144    self
145  }
146
147  /// 设置组聚合宽限期
148  /// Set group aggregation grace period
149  pub fn with_group_grace_period(mut self, grace: Duration) -> Self {

154  /// 获取任务类型
155  /// Get task type
156  pub fn get_type(&self) -> &str {
157    &self.task_type
158  }
159
160  /// 获取任务负载
161  /// Get task payload

163    &self.payload
164  }

166  /// 获取任务负载作为 JSON

169    serde_json::from_slice(&self.payload).map_err(Into::into)

230impl TaskInfo {
231  /// 从 Protocol Buffer 消息创建任务信息
232  /// Create task information from Protocol Buffer message
233  pub fn from_proto(
234    msg: &proto::TaskMessage,
235    state: TaskState,
236    next_process_at: Option<DateTime<Utc>>,
237    result: Option<Vec<u8>>,
238  ) -> Self {
239    Self {
240      id: msg.id.clone(),
241      queue: msg.queue.clone(),
242      task_type: msg.r#type.clone(),
243      payload: msg.payload.clone(),
244      state,
245      max_retry: msg.retry,
246      retried: msg.retried,
247      last_err: if msg.error_msg.is_empty() {
248        None
249      } else {
250        Some(msg.error_msg.clone())
251      },
252      last_failed_at: if msg.last_failed_at == 0 {
253        None
254      } else {
255        Some(DateTime::from_timestamp(msg.last_failed_at, 0).unwrap_or_default())
256      },
257      timeout: if msg.timeout == 0 {
258        None
259      } else {
260        Some(Duration::from_secs(msg.timeout as u64))
261      },
262      deadline: if msg.deadline == 0 {
263        None
264      } else {
265        Some(DateTime::from_timestamp(msg.deadline, 0).unwrap_or_default())
266      },
267      group: if msg.group_key.is_empty() {
268        None
269      } else {
270        Some(msg.group_key.clone())
271      },
272      next_process_at,    // 需要从其他地方获取
273      is_orphaned: false, // 需要从其他地方确定
274      retention: if msg.retention == 0 {
275        None
276      } else {
277        Some(Duration::from_secs(msg.retention as u64))
278      },
279      completed_at: if msg.completed_at == 0 {
280        None
281      } else {
282        Some(DateTime::from_timestamp(msg.completed_at, 0).unwrap_or_default())
283      },
284      result, // 需要从其他地方获取
285    }
286  }
287
288  /// 转换为 Protocol Buffer 消息
289  /// Convert to Protocol Buffer message
290  pub fn to_proto(&self) -> proto::TaskMessage {
291    proto::TaskMessage {
292      id: self.id.clone(),
293      r#type: self.task_type.clone(),
294      payload: self.payload.clone(),
295      queue: self.queue.clone(),
296      retry: self.max_retry,
297      retried: self.retried,
298      error_msg: self.last_err.clone().unwrap_or_default(),
299      last_failed_at: self.last_failed_at.map(|dt| dt.timestamp()).unwrap_or(0),
300      timeout: self.timeout.map(|d| d.as_secs() as i64).unwrap_or(0),
301      deadline: self.deadline.map(|dt| dt.timestamp()).unwrap_or(0),
302      unique_key: String::new(), // 需要单独计算
303      group_key: self.group.clone().unwrap_or_default(),
304      retention: self.retention.map(|d| d.as_secs() as i64).unwrap_or(0),
305      completed_at: self.completed_at.map(|dt| dt.timestamp()).unwrap_or(0),
306    }
307  }
308}
