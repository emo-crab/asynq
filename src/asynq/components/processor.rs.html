1//! 处理器模块
2//! Processor module
3//!
4//! 实现与 Go asynq processor.go 兼容的任务处理器
5//! Implements task processor compatible with Go asynq processor.go
6//!
7//! ## 概述 / Overview
8//!
9//! Processor 是任务处理的核心组件，负责从 Redis 队列中取出任务并执行。
10//! 它实现了与 Go 版本 asynq processor.go 相似的架构和功能。
11//!
12//! The Processor is the core component for task processing, responsible for dequeuing
13//! tasks from Redis queues and executing them. It implements an architecture and
14//! functionality similar to Go's asynq processor.go.
15//!
16//! ## 主要特性 / Key Features
17//!
18//! - **信号量并发控制**: 使用 Tokio Semaphore 限制并发工作者数量
19//!   - **Semaphore-based concurrency**: Uses Tokio Semaphore to limit concurrent workers
20//!
21//! - **队列优先级**: 支持严格优先级和加权优先级两种模式
22//!   - **Queue priority**: Supports both strict priority and weighted priority modes
23//!
24//! - **任务超时**: 支持任务级别和全局超时设置
25//!   - **Task timeout**: Supports task-level and global timeout settings
26//!
27//! - **优雅关闭**: 等待所有活跃工作者完成后再关闭
28//!   - **Graceful shutdown**: Waits for all active workers to finish before shutting down
29//!
30//! - **自动重试**: 失败任务自动重试，支持指数退避策略
31//!   - **Automatic retry**: Failed tasks are automatically retried with exponential backoff
32//!
33//! - **任务归档**: 达到最大重试次数后自动归档任务
34//!   - **Task archival**: Tasks are automatically archived after max retries
35//!
36//! ## 与 Go 版本的兼容性 / Compatibility with Go version
37//!
38//! 本实现与 Go 版本的 processor.go 在以下方面保持兼容：
39//! This implementation maintains compatibility with Go's processor.go in the following aspects:
40//!
41//! - 相同的队列优先级算法 / Same queue priority algorithm
42//! - 相同的重试延迟计算 / Same retry delay calculation
43//! - 相同的任务超时处理 / Same task timeout handling
44//! - 相同的并发控制机制 / Same concurrency control mechanism
45//!
46//! ## 使用示例 / Usage Example
47//!
48//! ```rust,no_run
49//! use asynq::components::processor::{Processor, ProcessorParams};
50//! use asynq::server::Handler;
51//! use std::collections::HashMap;
52//! use std::sync::Arc;
53//! use std::sync::atomic::AtomicUsize;
54//! use std::time::Duration;
55//! #[cfg(feature = "default")]
56//! async fn example() -> Result<(), Box<dyn std::error::Error>> {
57//! // 创建 broker 和其他必要组件
58//! // Create broker and other necessary components
59//! # use asynq::backend::RedisConnectionType;
60//! # use asynq::backend::RedisBroker;
61//! # let redis_config = asynq::backend::RedisConnectionType::single("redis://localhost:6379")?;
62//! # let broker = Arc::new(RedisBroker::new(redis_config).await?);
63//!
64//! let mut queues = HashMap::new();
65//! queues.insert("default".to_string(), 3);
66//! queues.insert("critical".to_string(), 6);
67//!
68//! let params = ProcessorParams {
69//!   broker: broker.clone(),
70//!   inspector: Arc::new(asynq::inspector::Inspector::from_broker(broker)),
71//!   queues,
72//!   concurrency: 10,
73//!   strict_priority: false,
74//!   task_check_interval: Duration::from_secs(1),
75//!   shutdown_timeout: Duration::from_secs(30),
76//!   active_workers: Arc::new(AtomicUsize::new(0)),

81//! // 启动处理器

123/// 任务取消追踪结构
124/// Task cancellation tracking structure
125///
126/// 对应 Go asynq 的 internal/base/base.go 中的 Cancellations
127/// Corresponds to Cancellations in Go asynq's internal/base/base.go
128#[derive(Clone)]
129pub struct CancellationMap {
130  // 正在运行的任务及其取消令牌
131  // Running tasks with their cancellation tokens
132  tasks: Arc<Mutex<HashMap<String, CancellationToken>>>,
133}
134
135impl CancellationMap {
136  /// 创建新的 Cancellations 实例
137  /// Create a new Cancellations instance
138  pub fn new() -> Self {
139    Self {
140      tasks: Arc::new(Mutex::new(HashMap::new())),
141    }
142  }
143
144  /// 添加任务取消令牌
145  /// Add task cancellation token
146  pub fn add(&self, task_id: String, token: CancellationToken) {
147    if let Ok(mut tasks) = self.tasks.lock() {
148      tasks.insert(task_id, token);
149    };
150  }
151
152  /// 移除任务取消令牌
153  /// Remove task cancellation token
154  pub fn remove(&self, task_id: &str) {
155    if let Ok(mut tasks) = self.tasks.lock() {
156      tasks.remove(task_id);
157    }
158  }
159
160  /// 取消指定的任务
161  /// Cancel specified task
162  ///
163  /// 对应 Go 的 cancelations.Cancel(taskID)
164  /// Corresponds to Go's cancelations.Cancel(taskID)
165  pub fn cancel(&self, task_id: &str) -> bool {
166    tracing::info!("canceling task {}", task_id);
167    if let Ok(tasks) = self.tasks.lock() {
168      if let Some(token) = tasks.get(task_id) {
169        token.cancel();
170        true
171      } else {
172        false
173      }
174    } else {
175      false
176    }
177  }
178  /// 获取正在运行的任务数量
179  /// Get the number of running tasks
180  pub fn len(&self) -> usize {
181    if let Ok(tasks) = self.tasks.lock() {
182      tasks.len()
183    } else {
184      0
185    }
186  }
187
188  /// 检查是否为空
189  /// Check if empty
190  pub fn is_empty(&self) -> bool {
191    self.len() == 0
192  }
193}

244  /// 创建新的处理器
245  /// Create a new processor
246  pub fn new(params: ProcessorParams) -> Self {
247    let queues = normalize_queues(params.queues);
248    let ordered_queues = if params.strict_priority {
249      Some(sort_by_priority(&queues))
250    } else {
251      None
252    };
253
254    let (quit_tx, quit_rx) = mpsc::channel(1);
255    let (abort_tx, _abort_rx) = mpsc::channel(1);
256
257    Self {
258      broker: params.broker,
259      inspector: params.inspector,
260      queue_config: queues,
261      ordered_queues,
262      task_check_interval: params.task_check_interval,
263      shutdown_timeout: params.shutdown_timeout,
264      sema: Arc::new(Semaphore::new(params.concurrency)),
265      running: Arc::new(AtomicBool::new(false)),
266      quit_tx: Some(quit_tx),
267      quit_rx: Some(quit_rx),
268      abort_tx: Some(abort_tx),
269      handle: None,
270      active_workers: params.active_workers,
271      cancellations: CancellationMap::new(),
272    }
273  }

284  /// 启动处理器
285  /// Start the processor
286  pub fn start<H>(&mut self, handler: Arc<H>)
287  where
288    H: Handler + 'static,
289  {
290    self.running.store(true, Ordering::SeqCst);
291
292    let broker = Arc::clone(&self.broker);
293    let inspector = Arc::clone(&self.inspector);
294    let running = Arc::clone(&self.running);
295    let sema = Arc::clone(&self.sema);
296    let queue_config = self.queue_config.clone();
297    let ordered_queues = self.ordered_queues.clone();
298    let task_check_interval = self.task_check_interval;
299    let active_workers = Arc::clone(&self.active_workers);
300    let cancelations = self.cancellations.clone();
301    if let Some(mut quit_rx) = self.quit_rx.take() {
302      let handle = tokio::spawn(async move {
303        loop {
304          // 检查是否收到退出信号
305          // Check if quit signal received
306          if quit_rx.try_recv().is_ok() {
307            tracing::debug!("Processor received quit signal");
308            break;
309          }
310
311          if !running.load(Ordering::SeqCst) {
312            break;
313          }
314
315          // 尝试获取信号量令牌
316          // Try to acquire semaphore permit
317          let permit = match sema.clone().try_acquire_owned() {
318            Ok(permit) => permit,
319            Err(_) => {
320              // 没有可用的工作者槽位，短暂等待
321              // No available worker slots, wait briefly
322              tokio::time::sleep(Duration::from_millis(100)).await;
323              continue;
324            }
325          };
326
327          // 获取队列列表
328          // Get queue list
329          let queues = get_queues(&queue_config, ordered_queues.as_ref());
330
331          // 从队列中取出任务
332          // Dequeue a task from the queue
333          match broker.dequeue(&queues).await {
334            Ok(Some(task_msg)) => {
335              // 增加活跃工作者计数
336              // Increment active worker count
337              active_workers.fetch_add(1, Ordering::Relaxed);
338
339              let handler = Arc::clone(&handler);
340              let broker = Arc::clone(&broker);
341              let inspector = Arc::clone(&inspector);
342              let active_workers = Arc::clone(&active_workers);
343              let cancelations = cancelations.clone();
344
345              // 在新的任务中处理
346              // Process in a new task
347              tokio::spawn(async move {
348                let _permit = permit; // 持有许可直到任务完成
349
350                // 创建任务，包含头部信息
351                // Create task with headers to preserve workflow path and other metadata
352                let mut task = match Task::new_with_headers(
353                  &task_msg.r#type,
354                  &task_msg.payload,
355                  task_msg.headers.clone(),
356                ) {
357                  Ok(task) => task,
358                  Err(e) => {
359                    tracing::error!("Failed to create task: {}", e);
360                    active_workers.fetch_sub(1, Ordering::Relaxed);
361                    return;
362                  }
363                };
364
365                // 创建并附加 ResultWriter
366                // Create and attach ResultWriter
367                // Note: ResultWriter is created for every task to match Go asynq behavior.
368                // The broker's write_result will only persist data when retention is configured.
369                let result_writer = Arc::new(crate::task::ResultWriter::new(
370                  task_msg.id.clone(),
371                  task_msg.queue.clone(),
372                  broker.clone(),
373                ));
374                task = task.with_result_writer(result_writer);
375                task = task.with_inspector(inspector);
376                // 创建取消令牌
377                // Create cancellation token
378                let cancel_token = CancellationToken::new();
379                let task_id = task_msg.id.clone();
380
381                // 注册取消令牌
382                // Register cancellation token
383                cancelations.add(task_id.clone(), cancel_token.clone());
384
385                // 计算任务超时
386                // Calculate task timeout
387                let timeout_duration = calculate_task_timeout(&task_msg);
388
389                // 执行任务，支持超时和取消
390                // Execute task with timeout and cancellation support
391                let result = if let Some(timeout) = timeout_duration {
392                  tokio::select! {
393                    // 任务执行
394                    // Task execution
395                    result = handler.process_task(task.clone()) => result,
396                    // 超时
397                    // Timeout
398                    _ = tokio::time::sleep(timeout) => {
399                      tracing::warn!("Task {} timed out after {:?}", task_id, timeout);
400                      Err(Error::other("Task execution timeout"))
401                    }
402                    // 取消信号
403                    // Cancellation signal
404                    _ = cancel_token.cancelled() => {
405                      tracing::info!("Task {} was cancelled", task_id);
406                      Ok(())
407                    }
408                  }
409                } else {
410                  tokio::select! {
411                    // 任务执行
412                    // Task execution
413                    result = handler.process_task(task.clone()) => result,
414                    // 取消信号
415                    // Cancellation signal
416                    _ = cancel_token.cancelled() => {
417                      tracing::info!("Task {} was cancelled", task_id);
418                      Err(Error::other("Task cancelled"))
419                    }
420                  }
421                };
422
423                // 移除取消令牌
424                // Remove cancellation token
425                cancelations.remove(&task_id);
426
427                // 处理结果
428                // Handle result
429                match result {
430                  Ok(()) => {
431                    // 任务成功
432                    // Task succeeded
433                    if task_msg.retention == 0 {
434                      if let Err(e) = broker.done(&task_msg).await {
435                        tracing::error!("Failed to mark task as done: {}", e);
436                      }
437                    } else if let Err(e) = broker.mark_as_complete(&task_msg).await {
438                      tracing::error!("Failed to mark task as complete: {}", e);
439                    }
440                  }
441                  Err(e) => {
442                    // 任务失败，决定重试还是归档
443                    // Task failed, decide retry or archive
444                    if task_msg.retried < task_msg.retry {
445                      // 计算重试延迟
446                      // Calculate retry delay
447                      let retry_delay =
448                        calculate_retry_delay(task_msg.retried, task.options.retry_policy.as_ref());
449                      let retry_at = chrono::Utc::now()
450                        + chrono::Duration::seconds(retry_delay.as_secs() as i64);
451
452                      if let Err(e) = broker.requeue(&task_msg, retry_at, &e.to_string()).await {
453                        tracing::error!("Failed to requeue task: {}", e);
454                      }
455                    } else {
456                      // 归档任务
457                      // Archive task
458                      if let Err(e) = broker.archive(&task_msg, &e.to_string()).await {
459                        tracing::error!("Failed to archive task: {}", e);
460                      }
461                    }
462                  }
463                }
464
465                // 减少活跃工作者计数
466                // Decrement active worker count
467                active_workers.fetch_sub(1, Ordering::Relaxed);
468              });
469            }
470            Ok(None) => {
471              // 没有任务，等待后重试
472              // No tasks, wait and retry
473              drop(permit); // 释放许可
474              tokio::time::sleep(task_check_interval).await;
475            }
476            Err(e) => {
477              // 出队错误
478              // Dequeue error
479              tracing::error!("Dequeue error: {}", e);
480              drop(permit); // 释放许可
481              tokio::time::sleep(Duration::from_secs(1)).await;
482            }
483          }
484        }
485
486        tracing::debug!("Processor loop exited");
487      });
488      self.handle = Some(handle);
489    } else {
490      self.handle = None;
491    };
492  }

494  /// 停止处理器（不等待工作者完成）
495  /// Stop the processor (without waiting for workers)
496  pub fn stop(&mut self) {
497    self.running.store(false, Ordering::SeqCst);
498    if let Some(tx) = self.quit_tx.take() {
499      let _ = tx.try_send(());
500    }
501  }
502
503  /// 关闭处理器并等待所有工作者完成
504  /// Shutdown the processor and wait for all workers to finish
505  pub async fn shutdown(&mut self) {
506    self.stop();
507
508    // 启动超时定时器，之后发送中止信号
509    // Start timeout timer, then send abort signal
510    let abort_tx = self.abort_tx.clone();
511    let shutdown_timeout = self.shutdown_timeout;
512    tokio::spawn(async move {
513      tokio::time::sleep(shutdown_timeout).await;
514      if let Some(tx) = abort_tx {
515        let _ = tx.send(()).await;
516      }
517    });
518
519    // 等待处理器主循环退出
520    // Wait for processor main loop to exit
521    if let Some(handle) = self.handle.take() {
522      let _ = handle.await;
523    }
524
525    tracing::info!("Waiting for all workers to finish...");
526
527    // 等待所有信号量令牌被释放（即所有工作者完成）
528    // Wait for all semaphore permits to be released (i.e., all workers finished)
529    let sema = Arc::clone(&self.sema);
530    let concurrency = sema.available_permits();
531    for _ in 0..concurrency {
532      let _ = sema.acquire().await;
533    }
534
535    tracing::info!("All workers have finished");
536  }

538/// 标准化队列配置，确保优先级为正数
539/// Normalize queue config, ensure priorities are positive
540fn normalize_queues(queues: HashMap<String, i32>) -> HashMap<String, i32> {
541  queues
542    .into_iter()
543    .map(|(name, priority)| (name, priority.max(1)))
544    .collect()
545}

547/// 按优先级排序队列（降序）
548/// Sort queues by priority (descending)
549fn sort_by_priority(queues: &HashMap<String, i32>) -> Vec<String> {
550  let mut queue_vec: Vec<_> = queues.iter().collect();
551  queue_vec.sort_by(|a, b| b.1.cmp(a.1)); // 降序排序
552  queue_vec
553    .into_iter()
554    .map(|(name, _)| name.clone())
555    .collect()
556}

558/// 获取队列列表，基于优先级
559/// Get queue list based on priority
560fn get_queues(
561  queue_config: &HashMap<String, i32>,
562  ordered_queues: Option<&Vec<String>>,
563) -> Vec<String> {
564  // 如果只有一个队列，直接返回
565  // If only one queue, return directly
566  if queue_config.len() == 1 {
567    return queue_config.keys().cloned().collect();
568  }
569
570  // 如果有排序的队列（严格优先级模式），返回排序后的队列
571  // If ordered queues exist (strict priority mode), return ordered queues
572  if let Some(ordered) = ordered_queues {
573    return ordered.clone();
574  }
575
576  // 否则，基于优先级加权随机选择
577  // Otherwise, weighted random selection based on priority
578  let mut names = Vec::new();
579  for (name, &priority) in queue_config {
580    for _ in 0..priority {
581      names.push(name.clone());
582    }
583  }
584
585  // 随机打乱
586  // Shuffle randomly
587  use rand::seq::SliceRandom;
588  let mut rng = rand::rng();
589  names.shuffle(&mut rng);
590
591  // 去重并返回
592  // Deduplicate and return
593  let mut seen = std::collections::HashSet::new();
594  let mut result = Vec::new();
595  for name in names {
596    if seen.insert(name.clone()) {
597      result.push(name);
598    }
599    if result.len() == queue_config.len() {
600      break;
601    }
602  }
603  result
604}

606/// 计算任务超时时间
607/// Calculate task timeout
608fn calculate_task_timeout(task_msg: &crate::proto::TaskMessage) -> Option<Duration> {
609  use crate::base::constants::DEFAULT_TIMEOUT;
610
611  // 优先使用任务级别的超时设置
612  // Prefer task-level timeout settings
613  if task_msg.timeout > 0 {
614    return Some(Duration::from_secs(task_msg.timeout as u64));
615  }
616
617  // 如果任务有截止时间，计算剩余时间作为超时
618  // If the task has a deadline, calculate remaining time as timeout
619  if task_msg.deadline > 0 {
620    let now = chrono::Utc::now().timestamp();
621    let remaining = task_msg.deadline - now;
622    if remaining > 0 {
623      return Some(Duration::from_secs(remaining as u64));
624    }
625  }
626
627  // 使用默认超时
628  // Use default timeout
629  Some(DEFAULT_TIMEOUT)
630}

632/// 计算重试延迟
633/// Calculate retry delay
634fn calculate_retry_delay(
635  retried: i32,
636  retry_policy: Option<&crate::backend::option::RetryPolicy>,
637) -> Duration {
638  match retry_policy {
639    Some(policy) => policy.calculate_delay(retried),
640    None => {
641      // 默认指数退避策略 - 与 Go 版本兼容
642      // Default exponential backoff strategy - compatible with Go version
643      let base_delay = (retried as f64).powf(4.0) as u64 + 15;
644      let jitter = rand::random::<u64>() % (30 * (retried as u64 + 1));
645      Duration::from_secs(base_delay + jitter)
646    }
647  }
648}
