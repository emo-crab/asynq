1//! 处理器模块
2//! Processor module
3//!
4//! 实现与 Go asynq processor.go 兼容的任务处理器
5//! Implements task processor compatible with Go asynq processor.go
6//!
7//! ## 概述 / Overview
8//!
9//! Processor 是任务处理的核心组件，负责从 Redis 队列中取出任务并执行。
10//! 它实现了与 Go 版本 asynq processor.go 相似的架构和功能。
11//!
12//! The Processor is the core component for task processing, responsible for dequeuing
13//! tasks from Redis queues and executing them. It implements an architecture and
14//! functionality similar to Go's asynq processor.go.
15//!
16//! ## 主要特性 / Key Features
17//!
18//! - **信号量并发控制**: 使用 Tokio Semaphore 限制并发工作者数量
19//!   - **Semaphore-based concurrency**: Uses Tokio Semaphore to limit concurrent workers
20//!
21//! - **队列优先级**: 支持严格优先级和加权优先级两种模式
22//!   - **Queue priority**: Supports both strict priority and weighted priority modes
23//!
24//! - **任务超时**: 支持任务级别和全局超时设置
25//!   - **Task timeout**: Supports task-level and global timeout settings
26//!
27//! - **优雅关闭**: 等待所有活跃工作者完成后再关闭
28//!   - **Graceful shutdown**: Waits for all active workers to finish before shutting down
29//!
30//! - **自动重试**: 失败任务自动重试，支持指数退避策略
31//!   - **Automatic retry**: Failed tasks are automatically retried with exponential backoff
32//!
33//! - **任务归档**: 达到最大重试次数后自动归档任务
34//!   - **Task archival**: Tasks are automatically archived after max retries
35//!
36//! ## 与 Go 版本的兼容性 / Compatibility with Go version
37//!
38//! 本实现与 Go 版本的 processor.go 在以下方面保持兼容：
39//! This implementation maintains compatibility with Go's processor.go in the following aspects:
40//!
41//! - 相同的队列优先级算法 / Same queue priority algorithm
42//! - 相同的重试延迟计算 / Same retry delay calculation
43//! - 相同的任务超时处理 / Same task timeout handling
44//! - 相同的并发控制机制 / Same concurrency control mechanism
45//!
46//! ## 使用示例 / Usage Example
47//!
48//! ```rust,no_run
49//! use asynq::components::processor::{Processor, ProcessorParams};
50//! use asynq::server::Handler;
51//! use std::collections::HashMap;
52//! use std::sync::Arc;
53//! use std::sync::atomic::AtomicUsize;
54//! use std::time::Duration;
55//!
56//! # async fn example() -> Result<(), Box<dyn std::error::Error>> {
57//! // 创建 broker 和其他必要组件
58//! // Create broker and other necessary components
59//! # use asynq::redis::RedisConnectionType;
60//! # use asynq::rdb::RedisBroker;
61//! # let redis_config = RedisConnectionType::single("redis://localhost:6379")?;
62//! # let broker = Arc::new(RedisBroker::new(redis_config).await?);
63//!
64//! let mut queues = HashMap::new();
65//! queues.insert("default".to_string(), 3);
66//! queues.insert("critical".to_string(), 6);
67//!
68//! let params = ProcessorParams {
69//!   broker: broker.clone(),
70//!   queues,
71//!   concurrency: 10,
72//!   strict_priority: false,
73//!   task_check_interval: Duration::from_secs(1),
74//!   shutdown_timeout: Duration::from_secs(30),
75//!   active_workers: Arc::new(AtomicUsize::new(0)),
76//! };
77//!
78//! let mut processor = Processor::new(params);
79//!
80//! // 启动处理器
81//! // Start processor

120/// 任务取消追踪结构
121/// Task cancellation tracking structure
122///
123/// 对应 Go asynq 的 internal/base/base.go 中的 Cancellations
124/// Corresponds to Cancellations in Go asynq's internal/base/base.go
125#[derive(Clone)]
126pub struct CancellationMap {
127  // 正在运行的任务及其取消令牌
128  // Running tasks with their cancellation tokens
129  tasks: Arc<Mutex<HashMap<String, CancellationToken>>>,
130}
131
132impl CancellationMap {
133  /// 创建新的 Cancellations 实例
134  /// Create a new Cancellations instance
135  pub fn new() -> Self {
136    Self {
137      tasks: Arc::new(Mutex::new(HashMap::new())),
138    }
139  }
140
141  /// 添加任务取消令牌
142  /// Add task cancellation token
143  pub fn add(&self, task_id: String, token: CancellationToken) {
144    if let Ok(mut tasks) = self.tasks.lock() {
145      tasks.insert(task_id, token);
146    };
147  }
148
149  /// 移除任务取消令牌
150  /// Remove task cancellation token
151  pub fn remove(&self, task_id: &str) {
152    if let Ok(mut tasks) = self.tasks.lock() {
153      tasks.remove(task_id);
154    }
155  }
156
157  /// 取消指定的任务
158  /// Cancel specified task
159  ///
160  /// 对应 Go 的 cancelations.Cancel(taskID)
161  /// Corresponds to Go's cancelations.Cancel(taskID)
162  pub fn cancel(&self, task_id: &str) -> bool {
163    tracing::info!("canceling task {}", task_id);
164    if let Ok(tasks) = self.tasks.lock() {
165      if let Some(token) = tasks.get(task_id) {
166        token.cancel();
167        true
168      } else {
169        false
170      }
171    } else {
172      false
173    }
174  }
175  /// 获取正在运行的任务数量
176  /// Get the number of running tasks
177  pub fn len(&self) -> usize {
178    if let Ok(tasks) = self.tasks.lock() {
179      tasks.len()
180    } else {
181      0
182    }
183  }
184
185  /// 检查是否为空
186  /// Check if empty
187  pub fn is_empty(&self) -> bool {
188    self.len() == 0
189  }
190}

195  }

208  // Semaphore to limit number of concurrent workers

240  /// 创建新的处理器
241  /// Create a new processor
242  pub fn new(params: ProcessorParams) -> Self {
243    let queues = normalize_queues(params.queues);
244    let ordered_queues = if params.strict_priority {
245      Some(sort_by_priority(&queues))
246    } else {
247      None
248    };
249
250    let (quit_tx, quit_rx) = mpsc::channel(1);
251    let (abort_tx, _abort_rx) = mpsc::channel(1);
252
253    Self {
254      broker: params.broker,
255      queue_config: queues,
256      ordered_queues,
257      task_check_interval: params.task_check_interval,
258      shutdown_timeout: params.shutdown_timeout,
259      sema: Arc::new(Semaphore::new(params.concurrency)),
260      running: Arc::new(AtomicBool::new(false)),
261      quit_tx: Some(quit_tx),
262      quit_rx: Some(quit_rx),
263      abort_tx: Some(abort_tx),
264      handle: None,
265      active_workers: params.active_workers,
266      cancellations: CancellationMap::new(),
267    }
268  }

273  /// 用于在服务器中接收取消事件后调用 cancel 方法
274  /// Used to call the cancel method after receiving cancellation events in the server
275  pub fn cancellations(&self) -> CancellationMap {
276    self.cancellations.clone()
277  }

279  /// 启动处理器
280  /// Start the processor
281  pub fn start<H>(&mut self, handler: Arc<H>)
282  where
283    H: Handler + 'static,
284  {
285    self.running.store(true, Ordering::SeqCst);
286
287    let broker = Arc::clone(&self.broker);
288    let running = Arc::clone(&self.running);
289    let sema = Arc::clone(&self.sema);
290    let queue_config = self.queue_config.clone();
291    let ordered_queues = self.ordered_queues.clone();
292    let task_check_interval = self.task_check_interval;
293    let active_workers = Arc::clone(&self.active_workers);
294    let cancelations = self.cancellations.clone();
295    if let Some(mut quit_rx) = self.quit_rx.take() {
296      let handle = tokio::spawn(async move {
297        loop {
298          // 检查是否收到退出信号
299          // Check if quit signal received
300          if quit_rx.try_recv().is_ok() {
301            tracing::debug!("Processor received quit signal");
302            break;
303          }
304
305          if !running.load(Ordering::SeqCst) {
306            break;
307          }
308
309          // 尝试获取信号量令牌
310          // Try to acquire semaphore permit
311          let permit = match sema.clone().try_acquire_owned() {
312            Ok(permit) => permit,
313            Err(_) => {
314              // 没有可用的工作者槽位，短暂等待
315              // No available worker slots, wait briefly
316              tokio::time::sleep(Duration::from_millis(100)).await;
317              continue;
318            }
319          };
320
321          // 获取队列列表
322          // Get queue list
323          let queues = get_queues(&queue_config, ordered_queues.as_ref());
324
325          // 从队列中取出任务
326          // Dequeue a task from the queue
327          match broker.dequeue(&queues).await {
328            Ok(Some(task_msg)) => {
329              // 增加活跃工作者计数
330              // Increment active worker count
331              active_workers.fetch_add(1, Ordering::Relaxed);
332
333              let handler = Arc::clone(&handler);
334              let broker = Arc::clone(&broker);
335              let active_workers = Arc::clone(&active_workers);
336              let cancelations = cancelations.clone();
337
338              // 在新的任务中处理
339              // Process in a new task
340              tokio::spawn(async move {
341                let _permit = permit; // 持有许可直到任务完成
342
343                // 创建任务
344                // Create task
345                let task = match Task::new(&task_msg.r#type, &task_msg.payload) {
346                  Ok(task) => task,
347                  Err(e) => {
348                    tracing::error!("Failed to create task: {}", e);
349                    active_workers.fetch_sub(1, Ordering::Relaxed);
350                    return;
351                  }
352                };
353
354                // 创建取消令牌
355                // Create cancellation token
356                let cancel_token = CancellationToken::new();
357                let task_id = task_msg.id.clone();
358
359                // 注册取消令牌
360                // Register cancellation token
361                cancelations.add(task_id.clone(), cancel_token.clone());
362
363                // 计算任务超时
364                // Calculate task timeout
365                let timeout_duration = calculate_task_timeout(&task_msg);
366
367                // 执行任务，支持超时和取消
368                // Execute task with timeout and cancellation support
369                let result = if let Some(timeout) = timeout_duration {
370                  tokio::select! {
371                    // 任务执行
372                    // Task execution
373                    result = handler.process_task(task.clone()) => result,
374                    // 超时
375                    // Timeout
376                    _ = tokio::time::sleep(timeout) => {
377                      tracing::warn!("Task {} timed out after {:?}", task_id, timeout);
378                      Err(Error::other("Task execution timeout"))
379                    }
380                    // 取消信号
381                    // Cancellation signal
382                    _ = cancel_token.cancelled() => {
383                      tracing::info!("Task {} was cancelled", task_id);
384                      Ok(())
385                    }
386                  }
387                } else {
388                  tokio::select! {
389                    // 任务执行
390                    // Task execution
391                    result = handler.process_task(task.clone()) => result,
392                    // 取消信号
393                    // Cancellation signal
394                    _ = cancel_token.cancelled() => {
395                      tracing::info!("Task {} was cancelled", task_id);
396                      Err(Error::other("Task cancelled"))
397                    }
398                  }
399                };
400
401                // 移除取消令牌
402                // Remove cancellation token
403                cancelations.remove(&task_id);
404
405                // 处理结果
406                // Handle result
407                match result {
408                  Ok(()) => {
409                    // 任务成功
410                    // Task succeeded
411                    if let Err(e) = broker.done(&task_msg).await {
412                      tracing::error!("Failed to mark task as done: {}", e);
413                    }
414                  }
415                  Err(e) => {
416                    // 任务失败，决定重试还是归档
417                    // Task failed, decide retry or archive
418                    if task_msg.retried < task_msg.retry {
419                      // 计算重试延迟
420                      // Calculate retry delay
421                      let retry_delay =
422                        calculate_retry_delay(task_msg.retried, task.options.retry_policy.as_ref());
423                      let retry_at = chrono::Utc::now()
424                        + chrono::Duration::seconds(retry_delay.as_secs() as i64);
425
426                      if let Err(e) = broker.requeue(&task_msg, retry_at, &e.to_string()).await {
427                        tracing::error!("Failed to requeue task: {}", e);
428                      }
429                    } else {
430                      // 归档任务
431                      // Archive task
432                      if let Err(e) = broker.archive(&task_msg, &e.to_string()).await {
433                        tracing::error!("Failed to archive task: {}", e);
434                      }
435                    }
436                  }
437                }
438
439                // 减少活跃工作者计数
440                // Decrement active worker count
441                active_workers.fetch_sub(1, Ordering::Relaxed);
442              });
443            }
444            Ok(None) => {
445              // 没有任务，等待后重试
446              // No tasks, wait and retry
447              drop(permit); // 释放许可
448              tokio::time::sleep(task_check_interval).await;
449            }
450            Err(e) => {
451              // 出队错误
452              // Dequeue error
453              tracing::error!("Dequeue error: {}", e);
454              drop(permit); // 释放许可
455              tokio::time::sleep(Duration::from_secs(1)).await;
456            }
457          }
458        }
459
460        tracing::debug!("Processor loop exited");
461      });
462      self.handle = Some(handle);
463    } else {
464      self.handle = None;
465    };
466  }

468  /// 停止处理器（不等待工作者完成）
469  /// Stop the processor (without waiting for workers)
470  pub fn stop(&mut self) {
471    self.running.store(false, Ordering::SeqCst);
472    if let Some(tx) = self.quit_tx.take() {
473      let _ = tx.try_send(());
474    }
475  }
476
477  /// 关闭处理器并等待所有工作者完成
478  /// Shutdown the processor and wait for all workers to finish
479  pub async fn shutdown(&mut self) {
480    self.stop();
481
482    // 启动超时定时器，之后发送中止信号
483    // Start timeout timer, then send abort signal
484    let abort_tx = self.abort_tx.clone();
485    let shutdown_timeout = self.shutdown_timeout;
486    tokio::spawn(async move {
487      tokio::time::sleep(shutdown_timeout).await;
488      if let Some(tx) = abort_tx {
489        let _ = tx.send(()).await;
490      }
491    });
492
493    // 等待处理器主循环退出
494    // Wait for processor main loop to exit
495    if let Some(handle) = self.handle.take() {
496      let _ = handle.await;
497    }
498
499    tracing::info!("Waiting for all workers to finish...");
500
501    // 等待所有信号量令牌被释放（即所有工作者完成）
502    // Wait for all semaphore permits to be released (i.e., all workers finished)
503    let sema = Arc::clone(&self.sema);
504    let concurrency = sema.available_permits();
505    for _ in 0..concurrency {
506      let _ = sema.acquire().await;
507    }
508
509    tracing::info!("All workers have finished");
510  }

513/// 标准化队列配置，确保优先级为正数
514/// Normalize queue config, ensure priorities are positive
515fn normalize_queues(queues: HashMap<String, i32>) -> HashMap<String, i32> {
516  queues
517    .into_iter()
518    .map(|(name, priority)| (name, priority.max(1)))
519    .collect()
520}
521
522/// 按优先级排序队列（降序）
523/// Sort queues by priority (descending)
524fn sort_by_priority(queues: &HashMap<String, i32>) -> Vec<String> {
525  let mut queue_vec: Vec<_> = queues.iter().collect();
526  queue_vec.sort_by(|a, b| b.1.cmp(a.1)); // 降序排序
527  queue_vec
528    .into_iter()
529    .map(|(name, _)| name.clone())
530    .collect()
531}

533/// 获取队列列表，基于优先级
534/// Get queue list based on priority
535fn get_queues(
536  queue_config: &HashMap<String, i32>,
537  ordered_queues: Option<&Vec<String>>,
538) -> Vec<String> {
539  // 如果只有一个队列，直接返回
540  // If only one queue, return directly
541  if queue_config.len() == 1 {
542    return queue_config.keys().cloned().collect();
543  }
544
545  // 如果有排序的队列（严格优先级模式），返回排序后的队列
546  // If ordered queues exist (strict priority mode), return ordered queues
547  if let Some(ordered) = ordered_queues {
548    return ordered.clone();
549  }
550
551  // 否则，基于优先级加权随机选择
552  // Otherwise, weighted random selection based on priority
553  let mut names = Vec::new();
554  for (name, &priority) in queue_config {
555    for _ in 0..priority {
556      names.push(name.clone());
557    }
558  }
559
560  // 随机打乱
561  // Shuffle randomly
562  use rand::seq::SliceRandom;
563  let mut rng = rand::rng();
564  names.shuffle(&mut rng);
565
566  // 去重并返回
567  // Deduplicate and return
568  let mut seen = std::collections::HashSet::new();
569  let mut result = Vec::new();
570  for name in names {
571    if seen.insert(name.clone()) {
572      result.push(name);
573    }
574    if result.len() == queue_config.len() {
575      break;
576    }
577  }
578  result
579}

581/// 计算任务超时时间
582/// Calculate task timeout
583fn calculate_task_timeout(task_msg: &crate::proto::TaskMessage) -> Option<Duration> {
584  use crate::base::constants::DEFAULT_TIMEOUT;
585
586  // 优先使用任务级别的超时设置
587  // Prefer task-level timeout settings
588  if task_msg.timeout > 0 {
589    return Some(Duration::from_secs(task_msg.timeout as u64));
590  }
591
592  // 如果任务有截止时间，计算剩余时间作为超时
593  // If the task has a deadline, calculate remaining time as timeout
594  if task_msg.deadline > 0 {
595    let now = chrono::Utc::now().timestamp();
596    let remaining = task_msg.deadline - now;
597    if remaining > 0 {
598      return Some(Duration::from_secs(remaining as u64));
599    }
600  }
601
602  // 使用默认超时
603  // Use default timeout
604  Some(DEFAULT_TIMEOUT)
605}

607/// 计算重试延迟
608/// Calculate retry delay
609fn calculate_retry_delay(
610  retried: i32,
611  retry_policy: Option<&crate::rdb::option::RetryPolicy>,
612) -> Duration {
613  match retry_policy {
614    Some(policy) => policy.calculate_delay(retried),
615    None => {
616      // 默认指数退避策略 - 与 Go 版本兼容
617      // Default exponential backoff strategy - compatible with Go version
618      let base_delay = (retried as f64).powf(4.0) as u64 + 15;
619      let jitter = rand::random::<u64>() % (30 * (retried as u64 + 1));
620      Duration::from_secs(base_delay + jitter)
621    }
622  }
623}
