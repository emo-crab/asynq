24/// Lifecycle trait - 组件生命周期管理接口
25/// Lifecycle trait - Component components management interface
26///
27/// 此 trait 定义了组件的基本生命周期操作：启动、关闭和状态检查
28/// This trait defines the basic components operations for components: start, shutdown, and state check
29///
30/// # 实现者 / Implementors
31///
32/// - [`Aggregator`](aggregator::Aggregator) - 聚合任务到组中进行批量处理
33/// - [`Forwarder`](forwarder::Forwarder) - 转发已到期的调度任务和重试任务
34/// - [`Healthcheck`](healthcheck::Healthcheck) - 执行健康检查
35/// - [`Heartbeat`](heartbeat::Heartbeat) - 发送心跳以维护服务器状态
36/// - [`Janitor`](janitor::Janitor) - 清理过期任务和死亡服务器
37/// - [`PeriodicTaskManager`](periodic_task_manager::PeriodicTaskManager) - 管理周期性任务
38/// - [`Recoverer`](recoverer::Recoverer) - 恢复孤儿任务
39/// - [`Subscriber`](subscriber::Subscriber) - 订阅任务事件
40///
41/// # 注意 / Note
42///
43/// [`Processor`](processor::Processor) 没有实现此 trait，因为它具有不同的接口：
44/// [`Processor`](processor::Processor) does not implement this trait because it has a different interface:
45/// - 需要泛型 Handler 参数 / Requires a generic Handler parameter
46/// - `start()` 方法接受 `&mut self` / `start()` method takes `&mut self`
47/// - `shutdown()` 是异步的 / `shutdown()` is async
48///
49/// # 示例 / Example
50///
51/// ```rust,no_run
52/// use asynq::components::ComponentLifecycle;
53/// use asynq::components::janitor::{Janitor, JanitorConfig};
54/// use std::sync::Arc;
55/// # use asynq::redis::RedisConnectionType;
56/// # use asynq::rdb::RedisBroker;
57///
58/// # async fn example() -> Result<(), Box<dyn std::error::Error>> {
59/// # let redis_config = RedisConnectionType::single("redis://localhost:6379")?;
60/// # let broker = Arc::new(RedisBroker::new(redis_config).await?);
61/// let janitor = Arc::new(Janitor::new(broker, JanitorConfig::default()));
62///
63/// // 启动组件
64/// // Start component
65/// let handle = janitor.clone().start();
66///
67/// // 检查状态
68/// // Check state
69/// assert!(!janitor.is_done());
70///
71/// // 关闭组件
72/// // Shutdown component
73/// janitor.shutdown();
74/// assert!(janitor.is_done());
75/// # Ok(())
76/// # }
77/// ```
78pub trait ComponentLifecycle {
79  /// 启动组件
80  /// Start the component
81  ///
82  /// 此方法启动组件的后台任务，返回一个 JoinHandle 用于等待任务完成
83  /// This method starts the component's background task, returning a JoinHandle to wait for completion
84  ///
85  /// # 返回 / Returns
86  ///
87  /// 返回一个 `JoinHandle<()>`，可用于等待组件任务完成
88  /// Returns a `JoinHandle<()>` that can be used to wait for the component task to complete
89  fn start(self: Arc<Self>) -> JoinHandle<()>;
90
91  /// 关闭组件
92  /// Shutdown the component
93  ///
94  /// 此方法发送关闭信号给组件，组件会在完成当前操作后停止
95  /// This method sends a shutdown signal to the component, which will stop after completing current operations
96  fn shutdown(&self);
97
98  /// 检查组件是否已完成
99  /// Check if the component is done
100  ///
101  /// # 返回 / Returns
102  ///
103  /// 如果组件已停止返回 `true`，否则返回 `false`
104  /// Returns `true` if the component has stopped, otherwise returns `false`
105  fn is_done(&self) -> bool;
106}

108#[cfg(test)]
109mod tests {
110  use super::*;
111  use std::sync::atomic::{AtomicBool, Ordering};
112
113  // 测试示例实现
114  // Test example implementation
115  struct TestComponent {
116    done: Arc<AtomicBool>,
117  }
118
119  impl ComponentLifecycle for TestComponent {
120    fn start(self: Arc<Self>) -> JoinHandle<()> {
121      tokio::spawn(async move {
122        loop {
123          if self.done.load(Ordering::Relaxed) {
124            break;
125          }
126          tokio::time::sleep(std::time::Duration::from_millis(10)).await;
127        }
128      })
129    }
130
131    fn shutdown(&self) {
132      self.done.store(true, Ordering::Relaxed);
133    }
134
135    fn is_done(&self) -> bool {
136      self.done.load(Ordering::Relaxed)
137    }
138  }
139
140  #[tokio::test]
141  async fn test_lifecycle_trait() {
142    let component = Arc::new(TestComponent {
143      done: Arc::new(AtomicBool::new(false)),
144    });
145
146    assert!(!component.is_done());
147
148    let handle = component.clone().start();
149
150    tokio::time::sleep(std::time::Duration::from_millis(20)).await;
151    component.shutdown();
152
153    assert!(component.is_done());
154    handle.await.unwrap();
155  }
