1//! Heartbeat 模块
2//! Heartbeat module
3//!
4//! 对应 Go 版本的 heartbeater（heartbeat.go）职责：
5//! Responsibilities corresponding to the Go version's heartbeater (heartbeat.go):
6//! 周期性写入 ServerInfo 以续租 / 汇报活跃 worker 数，并在关闭时清理服务器状态。
7//! Periodically writes ServerInfo to renew lease / report active worker count, and cleans up server state on shutdown。
8//!
9//! 此实现包含了 workers map 用于追踪当前节点正在处理的任务。
10//! This implementation includes a workers map to track tasks being processed by the current node.
11

24/// 静态元数据（不随心跳周期变化）
25/// Static metadata (does not change with heartbeat cycles)
26#[derive(Debug, Clone)]
27pub struct HeartbeatMeta {
28  pub host: String,
29  pub pid: i32,
30  pub server_uuid: String,
31  pub concurrency: usize,
32  pub queues: HashMap<String, i32>,
33  pub strict_priority: bool,
34  pub started: SystemTime,
35  /// ACL 租户名称（用于多租户隔离）
36  /// ACL tenant name (for multi-tenant isolation)
37  pub acl_tenant: Option<String>,
38}
39
40impl From<(&HeartbeatMeta, i32)> for ServerInfo {
41  fn from(value: (&HeartbeatMeta, i32)) -> Self {
42    let (meta, active_worker_count) = value;
43    ServerInfo {
44      host: meta.host.clone(),
45      pid: meta.pid,
46      server_id: meta.server_uuid.clone(),
47      concurrency: meta.concurrency as i32,
48      queues: meta.queues.clone(),
49      strict_priority: meta.strict_priority,
50      status: ServerState::Active.as_str().to_string(),
51      start_time: Some(prost_types::Timestamp::from(meta.started)),
52      active_worker_count,
53    }
54  }
55}

57/// 活跃 worker 信息（对应 Go 版本的 workerInfo）
58/// Active worker information (corresponds to Go version's workerInfo)
59///
60/// 包含正在处理任务的 worker 的详细信息
61/// Contains detailed information about a worker processing a task
62#[derive(Clone)]
63pub struct WorkerInfoEntry {
64  /// 正在处理的任务消息
65  /// The task message the worker is processing
66  pub msg: TaskMessage,
67  /// worker 开始处理任务的时间
68  /// The time the worker started processing the task
69  pub started: SystemTime,
70  /// worker 必须完成任务的截止时间
71  /// Deadline by which the worker must finish processing the task
72  pub deadline: SystemTime,
73}

75/// Worker 事件枚举（统一的事件类型）
76/// Worker event enum (unified event type)
77///
78/// 用于通过单一通道传递 worker 启动和完成事件
79/// Used to pass worker starting and finished events through a single channel
80#[allow(clippy::large_enum_variant)]
81#[derive(Clone)]
82pub enum WorkerEvent {
83  /// Worker 开始处理任务事件
84  /// Worker started processing task event
85  Started(WorkerInfoEntry),
86  /// Worker 完成任务事件（包含任务 ID）
87  /// Worker finished task event (contains task ID)
88  Finished(String),
89}

91/// 心跳器，封装心跳循环上下文
92/// Heartbeat, encapsulates heartbeat loop context
93///
94/// 对应 Go 版本的 heartbeater 结构体，包含 workers map 用于追踪正在处理的任务
95/// Corresponds to Go version's heartbeater struct, includes workers map to track tasks being processed
96pub struct Heartbeat {
97  broker: Arc<dyn Broker>,
98  interval: Duration,
99  meta: HeartbeatMeta,
100  /// 当前正在处理的任务 workers（对应 Go 的 workers map[string]*workerInfo）
101  /// Currently processing task workers (corresponds to Go's workers map[string]*workerInfo)
102  workers: RwLock<HashMap<String, WorkerInfoEntry>>,
103  shutting_down: Arc<AtomicBool>,
104  /// 接收 worker 事件的通道（统一接收启动和完成事件）
105  /// Channel to receive worker events (unified for starting and finished events)
106  event_rx: Mutex<Option<mpsc::Receiver<WorkerEvent>>>,
107}
108
109/// Worker 事件发送器（用于 Processor 发送事件给 Heartbeat）
110/// Worker event sender (for Processor to send events to Heartbeat)
111///
112/// 此结构实现了 Clone，支持多生产者模式
113/// This struct implements Clone, supporting multi-producer pattern
114#[derive(Clone)]
115pub struct WorkerEventSender {
116  /// 发送 worker 事件的通道（支持多生产者）
117  /// Channel to send worker events (supports multi-producer)
118  event_tx: mpsc::Sender<WorkerEvent>,
119}
120
121impl WorkerEventSender {
122  /// 发送 worker 开始事件
123  /// Send worker started event
124  pub async fn send_started(
125    &self,
126    entry: WorkerInfoEntry,
127  ) -> Result<(), mpsc::error::SendError<WorkerEvent>> {
128    self.event_tx.send(WorkerEvent::Started(entry)).await
129  }
130
131  /// 发送 worker 完成事件
132  /// Send worker finished event
133  pub async fn send_finished(
134    &self,
135    task_id: String,
136  ) -> Result<(), mpsc::error::SendError<WorkerEvent>> {
137    self.event_tx.send(WorkerEvent::Finished(task_id)).await
138  }
139}

141impl Heartbeat {
142  /// 创建新的心跳器
143  /// Create a new heartbeat
144  ///
145  /// 返回 (Heartbeat, WorkerEventSender)，其中 WorkerEventSender 用于 Processor 发送事件
146  /// Returns (Heartbeat, WorkerEventSender), where WorkerEventSender is used by Processor to send events
147  ///
148  /// WorkerEventSender 实现了 Clone，支持多生产者模式
149  /// WorkerEventSender implements Clone, supporting multi-producer pattern
150  pub fn new(
151    broker: Arc<dyn Broker>,
152    interval: Duration,
153    meta: HeartbeatMeta,
154  ) -> (Self, WorkerEventSender) {
155    // 创建统一的 worker 事件通道
156    // Create unified worker event channel
157    let (event_tx, event_rx) = mpsc::channel::<WorkerEvent>(100);
158
159    let heartbeat = Self {
160      broker,
161      interval,
162      meta,
163      workers: Default::default(),
164      shutting_down: Arc::new(AtomicBool::new(false)),
165      event_rx: Mutex::new(Some(event_rx)),
166    };
167
168    let sender = WorkerEventSender { event_tx };
169
170    (heartbeat, sender)
171  }
172
173  /// 启动心跳循环
174  /// Start the heartbeat loop
175  ///
176  /// 对应 Go 的 heartbeater.start()，同时处理心跳和 worker 事件
177  /// Corresponds to Go's heartbeater.start(), handles both heartbeat and worker events
178  ///
179  /// 如果 Heartbeat 已经启动过，返回一个立即完成的空任务
180  /// If Heartbeat has already been started, returns an immediately completing empty task
181  pub fn start(self: Arc<Self>) -> JoinHandle<()> {
182    tracing::info!("starting heartbeat");
183
184    let this = self;
185    tokio::spawn(async move {
186      // 取出 receiver（使用 Mutex 安全地获取）
187      // Take out receiver (safely obtain using Mutex)
188      let mut event_rx = {
189        let mut guard = this.event_rx.lock().await;
190        match guard.take() {
191          Some(rx) => rx,
192          None => {
193            tracing::warn!("Heartbeat already started, skipping duplicate start");
194            return;
195          }
196        }
197      };
198
199      let mut ticker = tokio::time::interval(this.interval);
200      loop {
201        tokio::select! {
202          // 定时心跳
203          // Periodic heartbeat
204          _ = ticker.tick() => {
205            if this.shutting_down.load(Ordering::Relaxed) {
206              break;
207            }
208            this.beat().await;
209          }
210          // 接收 worker 事件（统一处理启动和完成事件）
211          // Receive worker event (unified handling for starting and finished events)
212          Some(event) = event_rx.recv() => {
213            match event {
214              WorkerEvent::Started(worker_info) => {
215                let task_id = worker_info.msg.id.clone();
216                this.workers.write().await.insert(task_id.clone(), worker_info);
217                tracing::debug!("Worker started: task_id={}", task_id);
218              }
219              WorkerEvent::Finished(task_id) => {
220                this.workers.write().await.remove(&task_id);
221                tracing::debug!("Worker finished: task_id={}", task_id);
222              }
223            }
224          }
225        }
226      }
227      // 退出时清理服务器状态（冗余安全清理）
228      // Clean up server state on exit (redundant safe cleanup)
229      let _ = this
230        .broker
231        .clear_server_state(
232          &this.meta.host,
233          this.meta.pid,
234          &this.meta.server_uuid,
235          this.meta.acl_tenant.as_deref(),
236        )
237        .await;
238    })
239  }
240
241  /// 执行心跳（对应 Go 的 heartbeater.beat()）
242  /// Perform heartbeat (corresponds to Go's heartbeater.beat())
243  ///
244  /// 收集当前 worker 信息并写入 Redis
245  /// Collect current worker info and write to Redis
246  async fn beat(&self) {
247    let workers = self.workers.read().await;
248
249    // 构建 WorkerInfo 列表（对应 Go 的 ws []*base.WorkerInfo）
250    // Build WorkerInfo list (corresponds to Go's ws []*base.WorkerInfo)
251    let worker_infos: Vec<WorkerInfo> = workers
252      .values()
253      .map(|w| WorkerInfo {
254        host: self.meta.host.clone(),
255        pid: self.meta.pid,
256        server_id: self.meta.server_uuid.clone(),
257        task_id: w.msg.id.clone(),
258        task_type: w.msg.r#type.clone(),
259        task_payload: w.msg.payload.clone(),
260        queue: w.msg.queue.clone(),
261        start_time: Some(prost_types::Timestamp::from(w.started)),
262        deadline: Some(prost_types::Timestamp::from(w.deadline)),
263      })
264      .collect();
265
266    let active_worker_count = workers.len() as i32;
267    drop(workers); // 释放读锁 / Release read lock
268
269    let info: ServerInfo = (&self.meta, active_worker_count).into();
270
271    if let Err(e) = self
272      .broker
273      .write_server_state(
274        &info,
275        worker_infos,
276        self.interval * 2,
277        self.meta.acl_tenant.as_deref(),
278      )
279      .await
280    {
281      tracing::warn!("Heartbeat write failed: {}", e);
282    }
283  }

285  /// 请求心跳循环终止
286  /// Request the termination of the heartbeat loop
287  pub fn shutdown(&self) {
288    self.shutting_down.store(true, Ordering::Relaxed);
289  }
290
291  /// 检查是否已停止
292  /// Check if it has stopped
293  pub fn is_done(&self) -> bool {
294    self.shutting_down.load(Ordering::Relaxed)
295  }
296
297  /// 获取当前活跃 worker 数量
298  /// Get current active worker count
299  pub async fn active_worker_count(&self) -> usize {
300    self.workers.read().await.len()
301  }
302}
