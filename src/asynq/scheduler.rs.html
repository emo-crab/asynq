2//! # Periodic Task Scheduler (Scheduler)
3//!
4//! 该模块实现了类似 Go 版 asynq 的周期性任务调度功能。
5//! This module implements periodic task scheduling similar to the Go version of asynq.
6//! 主要参考 Go 版 asynq/scheduler.go、asynq/periodic_task.go。
7//! Mainly refers to Go asynq/scheduler.go and asynq/periodic_task.go.
8//!
9//! ## Go 版主要数据结构与方法对照：
10//! ## Main data structures and methods comparison with Go version:
11//!
12//! Go:
13//!   type Scheduler struct {
14//!       mu sync.Mutex
15//!       entries `map[string]*Entry`
16//!       running bool
17//!       ...
18//!   }
19//!   type Entry struct {
20//!       ID string
21//!       Spec string // cron 表达式
22//!       TaskType string
23//!       Payload []byte
24//!       Queue string
25//!       Next time.Time
26//!       ...
27//!   }
28//!   func (s *Scheduler) Register(spec, taskType string, payload []byte, queue string) error
29//!   func (s *Scheduler) Start()
30//!   func (s *Scheduler) Stop()
31//!   func (s *Scheduler) Remove(id string)
32//!   func (s *Scheduler) Entries() []*Entry
33//!
34//! Rust:
35//!   struct Scheduler { ... }
36//!   struct PeriodicTask { ... }
37//!   impl Scheduler {
38//!       pub fn add_task(&self, task: PeriodicTask)
39//!       pub fn remove_task(&self, name: &str)
40//!       pub fn list_tasks(&self) -> `Vec<String>`
41//!       pub fn start(&mut self)
42//!       pub fn stop(&mut self)
43//!   }
44//!
45//! 目前未实现的 Go 版功能：
46//! Unimplemented Go features:
47//!   - Entry ID（Rust 以 name 作为唯一标识）
48//!     - Entry ID (Rust uses name as unique identifier)
49//!   - 任务持久化与恢复
50//!     - Task persistence and recovery
51//!   - 任务去重与唯一性
52//!     - Task deduplication and uniqueness
53//!   - 运行中动态修改任务（可通过 add/remove 实现）
54//!     - Dynamic modification of tasks at runtime (can be done via add/remove)
55//!   - 任务执行日志与错误处理
56//!     - Task execution logging and error handling
57//!
58//! 如需补充上述功能，可参考 Go 版实现进一步扩展。
59//! For additional features, refer to the Go implementation for further extension。

77/// Represents a periodic task to be scheduled.
78#[derive(Debug, Clone)]
79pub struct PeriodicTask {
80  /// 任务名称，作为唯一标识
81  pub name: String,
82  /// cron 表达式
83  pub cron: String,
84  /// 任务负载
85  pub payload: Vec<u8>,
86  /// 任务队列
87  pub queue: String,
88  /// 任务选项
89  pub options: TaskOptions,
90  /// cron 调度对象
91  pub schedule: Schedule,
92  /// 下次执行时间
93  pub next_tick: Option<DateTime<Utc>>,
94}

97  /// 创建一个新的 PeriodicTask 实例
98  pub fn new(name: String, cron: String, payload: Vec<u8>, queue: String) -> anyhow::Result<Self> {
99    let schedule = Schedule::from_str(&cron)?;
100    let next_tick = schedule.upcoming(Utc).next();
101    let options = TaskOptions {

104    };
105    Ok(Self {
106      name,
107      cron,
108      payload,
109      queue,
110      options,
111      schedule,
112      next_tick,
113    })
114  }

141/// 周期性任务调度器
142pub struct Scheduler {
143  client: Arc<Client>,
144  /// 调度器唯一 id
145  id: String,
146  /// 存储所有任务的哈希表（entry_id -> PeriodicTask）
147  tasks: Arc<RwLock<HashMap<String, PeriodicTask>>>,
148  /// 运行状态标志
149  running: Arc<AtomicBool>,
150  /// 通知机制，用于唤醒调度器
151  notify: Arc<Notify>,
152  /// 后台运行的任务句柄
153  handles: SchedulerHandles,
154  /// 心跳间隔
155  heartbeat_interval: Duration,
156}

158impl Scheduler {
159  /// 初始化 Scheduler，自动生成 id，设置心跳间隔
160  pub async fn new(
161    client: Arc<Client>,
162    heartbeat_interval: Option<Duration>,
163  ) -> anyhow::Result<Self> {
164    let id = format!(
165      "{}:{}:{}",
166      hostname::get().unwrap_or_default().to_string_lossy(),
167      std::process::id(),
168      Uuid::new_v4()
169    );
170    Ok(Self {
171      client,
172      id,
173      tasks: Arc::new(RwLock::new(HashMap::new())),
174      running: Arc::new(AtomicBool::new(false)),
175      notify: Arc::new(Notify::new()),
176      handles: Arc::new(tokio::sync::Mutex::new(None)),
177      heartbeat_interval: heartbeat_interval.unwrap_or(Duration::from_secs(10)),
178    })
179  }

181  /// 添加一个周期性任务，并通过 rdb 写入 SchedulerEntry，兼容 Go 版 asynq
182  /// 注册一个周期性任务，只添加到本地，不写入 redis，兼容 Go 版 Register
183  pub async fn register(&self, mut task: PeriodicTask, queue: &str) -> anyhow::Result<String> {
184    let entry_id = Uuid::new_v4().to_string();
185    task.queue = queue.to_string();
186    task.options.queue = queue.to_string();
187    let schedule = Schedule::from_str(&task.cron)?;
188    task.schedule = schedule;
189    task.next_tick = task.schedule.upcoming(Utc).next();
190    let mut guard = self
191      .tasks
192      .write()
193      .map_err(|e| anyhow::anyhow!("lock poisoned: {e}"))?;
194    guard.insert(entry_id.clone(), task);
195    drop(guard);
196    self.notify.notify_one();
197    Ok(entry_id)
198  }
199
200  /// 注销/移除一个周期性任务（按 entry_id），只移除本地，不直接清理 redis
201  pub async fn unregister(&self, entry_id: &str) -> anyhow::Result<()> {
202    let mut guard = self
203      .tasks
204      .write()
205      .map_err(|e| anyhow::anyhow!("lock poisoned: {e}"))?;
206    guard.remove(entry_id);
207    drop(guard);
208    self.notify.notify_one();
209    Ok(())
210  }

220  /// 启动调度器，主循环和心跳分离
221  /// 这个方法现在是 pub(crate)，只能被 PeriodicTaskManager 调用
222  /// 在测试环境中也可以直接调用
223  #[cfg_attr(not(test), doc(hidden))]
224  pub async fn start(&self) {
225    // Prevent double start
226    if self.running.swap(true, Ordering::SeqCst) {
227      return;
228    }
229
230    let tasks = self.tasks.clone();
231    let running = self.running.clone();
232    let notify = self.notify.clone();
233    let client = self.client.clone();
234    let heartbeat_interval = self.heartbeat_interval;
235
236    let main_handle = tokio::spawn(async move {
237      loop {
238        if !running.load(Ordering::Relaxed) {
239          break;
240        }
241        let now = Utc::now();
242        let mut min_next: Option<DateTime<Utc>> = None;
243        let mut due_entries = Vec::new();
244        {
245          if let Ok(mut tasks) = tasks.write() {
246            for (entry_id, task) in tasks.iter_mut() {
247              let next_tick = task.next_tick;
248              if let Some(next) = next_tick {
249                if next <= now {
250                  due_entries.push((
251                    entry_id.clone(),
252                    task.name.clone(),
253                    task.payload.clone(),
254                    task.options.clone(),
255                  ));
256                  task.next_tick = task.schedule.upcoming(Utc).next();
257                }
258                if min_next.is_none() || min_next.map(|m| next < m).unwrap_or(false) {
259                  min_next = Some(next);
260                }
261              }
262            }
263          }
264        }
265        // 推送到期任务，并通过 broker 记录调度事件
266        for (entry_id, name, payload, options) in due_entries {
267          if let Ok(mut t) = Task::new(&name, &payload) {
268            t.options = options;
269            let _ = client.enqueue(t).await;
270            // 记录调度事件
271            let event = SchedulerEnqueueEvent {
272              task_id: name.clone(),
273              enqueue_time: Some(prost_types::Timestamp {
274                seconds: now.timestamp(),
275                nanos: now.timestamp_subsec_nanos() as i32,
276              }),
277            };
278            let broker = client.get_broker();
279            let entry_id = entry_id.clone();
280            tokio::spawn(async move {
281              let _ = broker
282                .record_scheduler_enqueue_event(&event, &entry_id)
283                .await;
284            });
285          }
286        }
287        let sleep_dur = min_next
288          .map(|t| (t - now).to_std().unwrap_or(Duration::from_secs(1)))
289          .unwrap_or(heartbeat_interval);
290        tokio::select! {
291            _ = tokio::time::sleep(sleep_dur) => {},
292            _ = notify.notified() => {},
293        }
294      }
295    });
296
297    // 启动心跳循环
298    let heartbeat_handle = self.spawn_heartbeat();
299
300    // Store handles
301    let mut handles_guard = self.handles.lock().await;
302    *handles_guard = Some((main_handle, heartbeat_handle));
303  }

305  /// 心跳循环，定期写入所有 entry 到 redis
306  fn spawn_heartbeat(&self) -> JoinHandle<()> {
307    let tasks = self.tasks.clone();
308    let running = self.running.clone();
309    let client = self.client.clone();
310    let scheduler_id = self.id.clone();
311    let heartbeat_interval = self.heartbeat_interval;
312    tokio::spawn(async move {
313      let broker = client.get_broker();
314      let mut ticker = tokio::time::interval(heartbeat_interval);
315      loop {
316        tokio::select! {
317          _ = ticker.tick() => {
318            if !running.load(Ordering::Relaxed) {
319              break;
320            }
321            // 快照写入 redis
322            let mut all_entries = Vec::new();
323            {
324              if let Ok(tasks) = tasks.read() {
325                for (entry_id, task) in tasks.iter() {
326                  let next_tick = task.next_tick;
327                  let entry = SchedulerEntry {
328                    id: entry_id.clone(),
329                    spec: task.cron.clone(),
330                    task_type: task.name.clone(),
331                    task_payload: task.payload.clone(),
332                    enqueue_options: Self::stringify_options(&task.options),
333                    next_enqueue_time: next_tick.map(|t| prost_types::Timestamp {
334                      seconds: t.timestamp(),
335                      nanos: t.timestamp_subsec_nanos() as i32,
336                    }),
337                    prev_enqueue_time: None,
338                  };
339                  all_entries.push(entry);
340                }
341              }
342            }
343            let _ = broker.write_scheduler_entries(&all_entries, &scheduler_id, (heartbeat_interval * 2).as_secs()).await;
344          }
345          _ = async {
346            while running.load(Ordering::Relaxed) {
347              tokio::time::sleep(heartbeat_interval).await;
348            }
349          } => {
350            // 清理 redis
351            let _ = broker.clear_scheduler_entries(&scheduler_id).await;
352            break;
353          }
354        }
355      }
356    })
357  }

359  /// 查询所有注册的 SchedulerEntry，兼容 Go 版 asynq Inspector
360  pub async fn list_entries(&self, scheduler_id: &str) -> Vec<SchedulerEntry> {
361    let broker = self.client.get_broker();
362    let raw_map = broker
363      .scheduler_entries_script(scheduler_id)
364      .await
365      .unwrap_or_default();
366    let mut entries = Vec::new();
367    for (_id, bytes) in raw_map {
368      if let Ok(entry) = SchedulerEntry::decode(&*bytes) {
369        entries.push(entry);
370      }
371    }
372    entries
373  }

391  /// 停止调度器并清理 redis entry
392  /// 这个方法现在是 pub(crate)，只能被 PeriodicTaskManager 调用
393  /// 在测试环境中也可以直接调用
394  #[cfg_attr(not(test), doc(hidden))]
395  pub async fn stop(&self) {
396    self.running.store(false, Ordering::SeqCst);
397    self.notify.notify_one();
398
399    // Take the handles and wait for them
400    let handles = {
401      let mut handles_guard = self.handles.lock().await;
402      handles_guard.take()
403    };
404
405    if let Some((main_handle, heartbeat_handle)) = handles {
406      let _ = main_handle.await;
407      let _ = heartbeat_handle.await;
408    }
409
410    // 正确清理 redis 中的 entry（只需用 scheduler_id）
411    let broker = self.client.get_broker();
412    let _ = broker.clear_scheduler_entries(&self.id).await;
413  }

415  /// 生成与 Go 版一致的调度选项字符串
416  /// Generate option strings compatible with Go asynq scheduler
417  pub fn stringify_options(opts: &TaskOptions) -> Vec<String> {
418    let option_types: Vec<OptionType> = opts.into();
419    option_types.iter().map(|opt| opt.to_string()).collect()
420  }
421
422  /// 解析选项字符串为 TaskOptions
423  /// Parse option strings to TaskOptions
424  pub fn parse_options(opts: &[String]) -> TaskOptions {
425    let mut option_types = Vec::new();
426    for opt_str in opts {
427      if let Ok(opt) = OptionType::parse(opt_str) {
428        option_types.push(opt);
429      }
430    }
431    option_types.into()
432  }
