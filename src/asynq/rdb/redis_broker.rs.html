30  /// 从RedisConnection创建新的Redis经纪人实例
31  /// Create a new Redis broker instance from RedisConnection
32  pub async fn new(conn: RedisConnectionType) -> Result<Self> {
33    match conn {
34      RedisConnectionType::Single(config) => {
35        let client = Client::open(config)?;
36        let mut broker = Self {
37          client: RedisClient::Single(client),
38          script_manager: ScriptManager::default(),
39        };
40        broker.init_scripts().await?;
41        Ok(broker)
42      }
43      #[cfg(feature = "cluster")]
44      RedisConnectionType::Cluster(config) => {
45        // 如果配置了使用 RESP3，创建 push_sender 通道
46        // If RESP3 is configured, create push_sender channel
47        let (push_receiver, cluster_client) = {
48          let (tx, rx) = tokio::sync::mpsc::unbounded_channel();
49          let client = redis::cluster::ClusterClientBuilder::new(config)
50            .use_protocol(redis::ProtocolVersion::RESP3)
51            .push_sender(tx)
52            .build()?;
53          (
54            std::sync::Arc::new(tokio::sync::Mutex::new(Some(rx))),
55            client,
56          )
57        };
58        let mut broker = Self {
59          client: RedisClient::Cluster {
60            client: cluster_client,
61            push_receiver,
62          },
63          script_manager: ScriptManager::default(),
64        };
65        broker.init_scripts().await?;
66        Ok(broker)
67      }
68    }
69  }

78  /// 获取 PubSub 连接
79  /// Get PubSub connection
80  pub async fn get_pubsub(&self) -> Result<RedisPubSub> {
81    match &self.client {
82      RedisClient::Single(client) => {
83        let pubsub = client.get_async_pubsub().await?;
84        Ok(RedisPubSub::Single(pubsub))
85      }
86      #[cfg(feature = "cluster")]
87      RedisClient::Cluster {
88        client,
89        push_receiver,
90      } => {
91        // 对于集群模式，从 push_receiver 取出接收器并获取连接
92        // For cluster mode, take receiver from push_receiver and get connection
93        let mut receiver_option = push_receiver.lock().await;
94        if let Some(receiver) = receiver_option.take() {
95          // 获取一个集群连接用于订阅
96          // Get a cluster connection for subscription
97          let connection = client.get_async_connection().await?;
98          let cluster_pubsub = ClusterPubSubConnection::from_receiver(receiver, connection);
99          Ok(RedisPubSub::Cluster(cluster_pubsub))
100        } else {
101          Err(Error::other(
102            "Cluster PubSub receiver has already been taken. You can only create one PubSub connection for cluster mode."
103          ))
104        }
105      }
106    }
107  }

127  pub fn decode_task_message(&self, data: &[u8]) -> Result<TaskMessage> {
128    match TaskMessage::decode(data) {

130      // 不要处理这里的错误，直接返回错误就可以了

133    }

137  /// Create TaskMessage from Task
138  pub(crate) fn task_to_message(&self, task: &Task) -> TaskMessage {
139    TaskMessage {
140      r#type: task.task_type.clone(),
141      payload: task.payload.clone(),
142      id: task
143        .options
144        .task_id

146        .unwrap_or(Uuid::new_v4().to_string()),
147      queue: if task.options.queue.clone().is_empty() {
148        DEFAULT_QUEUE_NAME.to_string()
149      } else {
150        task.options.queue.clone()
151      },
152      retry: task.options.max_retry,
153      retried: 0,
154      error_msg: String::new(),
155      last_failed_at: 0,
156      timeout: task
157        .options

159        .map(|d| d.as_secs() as i64)
160        .unwrap_or(0),
161      deadline: task.options.deadline.map(|d| d.timestamp()).unwrap_or(0),
162      unique_key: String::new(),
163      group_key: task.options.group.clone().unwrap_or_default(),
164      retention: task
165        .options
166        .retention
167        .map(|d| d.as_secs() as i64)
168        .unwrap_or(0),
169      completed_at: 0,
170    }

173  /// 批量写入 scheduler entries，兼容 Go 版 asynq
174  /// Batch write scheduler entries, compatible with Go version asynq
175  pub async fn write_scheduler_entries(
176    &self,
177    entries: &[SchedulerEntry],
178    scheduler_id: &str,
179    ttl_secs: u64,
180  ) -> Result<()> {
181    let mut args: Vec<RedisArg> = Vec::new();
182    args.push(RedisArg::Int(ttl_secs as i64));
183    for entry in entries {
184      let mut buf = Vec::new();
185      entry
186        .encode(&mut buf)
187        .map_err(|e| Error::other(format!("prost encode error: {e}")))?;
188      args.push(RedisArg::Bytes(buf));
189    }
190    let key = keys::scheduler_entries_key(scheduler_id);
191    let mut conn = self.get_async_connection().await?;
192    let _: () = self
193      .script_manager
194      .eval_script(
195        &mut conn,
196        "write_scheduler_entries",
197        std::slice::from_ref(&key),
198        &args,
199      )
200      .await?;
201    // 新增：ZADD 到全局 ZSET，score 为过期时间戳
202    // New: ZADD to global ZSET, score is the expiration timestamp
203    let zset_key = keys::ALL_SCHEDULERS;
204    let expire_at = chrono::Utc::now().timestamp() + ttl_secs as i64;
205    let _: () = conn.zadd(zset_key, &key, expire_at).await?;
206
207    Ok(())
208  }

210  /// 记录调度事件，兼容 Go 版 asynq
211  /// Record scheduling event, compatible with Go version asynq
212  pub async fn record_scheduler_enqueue_event(
213    &self,
214    event: &SchedulerEnqueueEvent,
215    entry_id: &str,
216  ) -> Result<()> {
217    let mut buf = Vec::new();
218    event
219      .encode(&mut buf)
220      .map_err(|e| Error::other(format!("prost encode error: {e}")))?;
221    let key = keys::scheduler_history_key(entry_id);
222    let mut conn = self.get_async_connection().await?;
223    let args = vec![
224      RedisArg::Int(event.enqueue_time.map(|x| x.seconds).unwrap_or(0)),
225      RedisArg::Bytes(buf),
226      RedisArg::Int(1000),
227    ];
228    self
229      .script_manager
230      .eval_script::<()>(&mut conn, "record_scheduler_enqueue_event", &[key], &args)
231      .await?;
232    Ok(())
233  }

235  /// 通过脚本获取所有 SchedulerEntry，兼容 Go 版 asynq
236  /// Get all SchedulerEntry through script, compatible with Go version asynq
237  pub async fn scheduler_entries_script(
238    &self,
239    scheduler_id: &str,
240  ) -> Result<std::collections::HashMap<String, Vec<u8>>> {
241    let key = keys::scheduler_entries_key(scheduler_id);
242    let mut conn = self.get_async_connection().await?;
243    // 调用脚本，假设脚本返回 [key1, value1, key2, value2, ...]
244    // Call the script, assuming the script returns [key1, value1, key2, value2, ...]
245    let result: Vec<Vec<u8>> = self
246      .script_manager
247      .eval_script(&mut conn, "get_scheduler_entries", &[key], &[])
248      .await?;
249    let mut map = std::collections::HashMap::new();
250    let mut iter = result.chunks_exact(2);
251    while let Some([k, v]) = iter.next() {
252      let key_str = String::from_utf8_lossy(k).to_string();
253      map.insert(key_str, v.clone());
254    }
255    Ok(map)
256  }

271  /// 删除 scheduler entries 数据，兼容 Go 版 asynq
272  /// Delete scheduler entries data, compatible with Go version asynq
273  pub async fn clear_scheduler_entries(&self, scheduler_id: &str) -> Result<()> {
274    let key = keys::scheduler_entries_key(scheduler_id);
275    let zset_key = keys::ALL_SCHEDULERS;
276    let mut conn = self.get_async_connection().await?;
277    // ZREM 全局调度器 ZSET
278    // ZREM global scheduler ZSET
279    let _: () = conn
280      .zrem(zset_key, &key)
281      .await
282      .map_err(|e| Error::other(format!("redis ZREM error: {e}")))?;
283    // DEL 具体 entry 列表
284    // DEL specific entry list
285    let _: () = conn
286      .del(&key)
287      .await
288      .map_err(|e| Error::other(format!("redis DEL error: {e}")))?;
289    Ok(())
290  }
