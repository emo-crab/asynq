22/// 客户端后端类型
23/// Client backend type
24enum ClientBroker {
25  /// Redis 后端
26  /// Redis backend
27  Redis(Arc<RedisBroker>),
28  #[cfg(feature = "postgres")]
29  /// PostgresSQL 后端
30  /// PostgresSQL backend
31  Postgres(Arc<PostgresBroker>),
32  #[cfg(feature = "websocket")]
33  /// WebSocket 后端（连接到 asynq-server）
34  /// WebSocket backend (connects to asynq-server)
35  WebSocket(Arc<WebSocketBroker>),
36}
37
38impl ClientBroker {
39  /// 获取 Broker trait 对象
40  /// Get the Broker trait object
41  fn as_broker(&self) -> Arc<dyn Broker> {
42    match self {
43      ClientBroker::Redis(broker) => broker.clone(),
44      #[cfg(feature = "postgres")]
45      ClientBroker::Postgres(broker) => broker.clone(),
46      #[cfg(feature = "websocket")]
47      ClientBroker::WebSocket(broker) => broker.clone(),
48    }
49  }
50
51  /// 获取 SchedulerBroker trait 对象
52  /// Get the SchedulerBroker trait object
53  fn as_scheduler_broker(&self) -> Arc<dyn crate::base::SchedulerBroker> {
54    match self {
55      ClientBroker::Redis(broker) => broker.clone(),
56      #[cfg(feature = "postgres")]
57      ClientBroker::Postgres(broker) => broker.clone(),
58      #[cfg(feature = "websocket")]
59      ClientBroker::WebSocket(broker) => broker.clone(),
60    }
61  }
62}

180  /// 应用 ACL 前缀到任务的队列名称
181  /// Apply ACL prefix to task's queue name
182  fn apply_acl_prefix_to_task(&self, mut task: Task) -> Task {
183    if self.config.acl_tenant.is_some() {
184      let queue = task.get_queue();
185      let prefixed_queue = self.config.get_queue_name_with_prefix(queue);
186      task = task.with_queue(prefixed_queue);
187    }
188    task
189  }
190
191  /// 将任务加入队列立即处理
192  /// Enqueue a task for immediate processing
193  pub async fn enqueue(&self, task: Task) -> Result<TaskInfo> {
194    let task = self.apply_acl_prefix_to_task(task);
195    self.broker.as_broker().enqueue(&task).await
196  }

205  /// 调度任务在指定时间处理
206  /// Schedule a task for processing at a specific time
207  pub async fn schedule(&self, task: Task, process_at: SystemTime) -> Result<TaskInfo> {
208    let task = self.apply_acl_prefix_to_task(task);
209    let date_time = chrono::DateTime::<chrono::Utc>::from(process_at);
210    self.broker.as_broker().schedule(&task, date_time).await
211  }
212
213  /// 调度唯一任务在指定时间处理
214  /// Schedule a unique task for processing at a specific time
215  pub async fn schedule_unique(
216    &self,
217    task: Task,
218    process_at: SystemTime,
219    ttl: Duration,
220  ) -> Result<TaskInfo> {
221    let task = self.apply_acl_prefix_to_task(task);
222    let date_time = chrono::DateTime::<chrono::Utc>::from(process_at);
223    self
224      .broker
225      .as_broker()
226      .schedule_unique(&task, date_time, ttl)
227      .await
228  }
229
230  /// 在指定延迟后处理任务
231  /// Enqueue a task to be processed after a specific delay
232  pub async fn enqueue_in(&self, task: Task, delay: Duration) -> Result<TaskInfo> {
233    let task = self.apply_acl_prefix_to_task(task);
234    let process_at = SystemTime::now()
235      .checked_add(delay)
236      .ok_or_else(|| std::io::Error::new(std::io::ErrorKind::InvalidData, "Delay overflow"))?;
237    self.schedule(task, process_at).await
238  }
