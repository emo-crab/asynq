1/// Pagination specifies the page size and page number for list operations.
2/// 分页结构体，指定列表操作的页面大小和页面编号
3#[derive(Debug, Clone, Copy)]
4pub struct Pagination {
5  /// Number of items in the page.
6  /// 每页的项目数
7  pub size: i64,
8  /// Page number starting from zero.
9  /// 从零开始的页面编号
10  pub page: i64,
11}
12
13impl Pagination {
14  /// Returns the start index for the current page.
15  /// 返回当前页面的起始索引
16  pub fn start(&self) -> i64 {
17    self.size * self.page
18  }
19  /// Returns the stop index for the current page.
20  /// 返回当前页面的结束索引
21  pub fn stop(&self) -> i64 {
22    self.size * self.page + self.size - 1
23  }
24}
25impl Default for Pagination {
26  fn default() -> Self {
27    Pagination { size: 20, page: 0 }
28  }
29}
