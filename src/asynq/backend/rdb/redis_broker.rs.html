32  /// 从RedisConnection创建新的Redis经纪人实例
33  /// Create a new Redis broker instance from RedisConnection
34  pub async fn new(conn: RedisConnectionType) -> Result<Self> {
35    match conn {
36      RedisConnectionType::Single {
37        connection_info,
38        #[cfg(feature = "tls")]
39        tls_certs,
40      } => {
41        // If a tls_mode was provided in RedisConnectionType, apply it to the
42        // ConnectionInfo addr so redis::Client will create a TLS connection.
43        // This is a no-op when the `tls` feature is not enabled.
44        let client = {
45          #[cfg(feature = "tls")]
46          {
47            // When tls feature is enabled, if a tls_mode was supplied by the
48            // higher-level RedisConnectionType, prefer building a TLS-enabled
49            // client using the redis crate's TLS helpers. Otherwise fall back
50            // to the normal Client::open.
51            let ci = connection_info.clone();
52            if let Some(tls) = tls_certs {
53              // Attempt to build a client with TLS configuration provided by
54              // callers (they can construct a `redis::TlsCertificates` in examples).
55              // This uses the redis crate API to attach rustls-based cert/key
56              // and optional root CA bytes to the client.
57              Client::build_with_tls(ci, tls)?
58            } else {
59              Client::open(ci)?
60            }
61          }
62          #[cfg(not(feature = "tls"))]
63          {
64            Client::open(connection_info)?
65          }
66        };
67        let mut broker = Self {
68          client: RedisClient::Single(client),
69          script_manager: ScriptManager::default(),
70        };
71        broker.init_scripts().await?;
72        Ok(broker)
73      }
74      #[cfg(feature = "cluster")]
75      crate::backend::RedisConnectionType::Cluster(config) => {
76        // 如果配置了使用 RESP3，创建 push_sender 通道
77        // If RESP3 is configured, create push_sender channel
78        let (push_receiver, cluster_client) = {
79          let (tx, rx) = tokio::sync::mpsc::unbounded_channel();
80          let client = redis::cluster::ClusterClientBuilder::new(config)
81            .use_protocol(redis::ProtocolVersion::RESP3)
82            .push_sender(tx)
83            .build()?;
84          (
85            std::sync::Arc::new(tokio::sync::Mutex::new(Some(rx))),
86            client,
87          )
88        };
89        let mut broker = Self {
90          client: RedisClient::Cluster {
91            client: cluster_client,
92            push_receiver,
93          },
94          script_manager: ScriptManager::default(),
95        };
96        broker.init_scripts().await?;
97        Ok(broker)
98      }
99      #[cfg(feature = "sentinel")]
100      crate::backend::RedisConnectionType::Sentinel {
101        master_name,
102        sentinels,
103        redis_connection_info,
104        #[cfg(feature = "tls")]
105        tls_certs,
106      } => {
107        // Use redis::SentinelClient to resolve the master and get a Client.
108        // See https://docs.rs/redis/latest/redis/sentinel/index.html
109        // Build a SentinelClient and wrap it in Arc<Mutex<>> so we can call its &mut async methods
110        // If any sentinel ConnectionInfo contains a username/password, use that to populate
111        // the SentinelNodeConnectionInfo so the resolved master connection will use auth.
112        let mut node_conn_info: Option<SentinelNodeConnectionInfo> = None;
113        if let Some(conn_info) = redis_connection_info {
114          let sentinel_node_connection_info =
115            SentinelNodeConnectionInfo::default().set_redis_connection_info(conn_info);
116          node_conn_info = Some(sentinel_node_connection_info);
117        }
118        // If TLS certificates are provided, use SentinelClientBuilder to attach them
119        #[cfg(feature = "tls")]
120        let sentinel_client = {
121          if let Some(certs) = tls_certs {
122            // SentinelClientBuilder expects ConnectionAddr values for sentinels
123            let addrs: Vec<redis::ConnectionAddr> =
124              sentinels.iter().map(|ci| ci.addr().clone()).collect();
125            let mut builder = redis::sentinel::SentinelClientBuilder::new(
126              addrs,
127              master_name.clone(),
128              SentinelServerType::Master,
129            )?;
130            builder = builder.set_client_to_redis_certificates(certs);
131            builder.build()?
132          } else {
133            // No certs provided; fall back to simple builder
134            SentinelClient::build(
135              sentinels,
136              master_name.clone(),
137              node_conn_info,
138              SentinelServerType::Master,
139            )?
140          }
141        };
142        #[cfg(not(feature = "tls"))]
143        let sentinel_client = {
144          SentinelClient::build(
145            sentinels,
146            master_name.clone(),
147            node_conn_info,
148            SentinelServerType::Master,
149          )?
150        };
151        let client = std::sync::Arc::new(tokio::sync::Mutex::new(sentinel_client));
152        let mut broker = Self {
153          client: RedisClient::Sentinel { client },
154          script_manager: ScriptManager::default(),
155        };
156        broker.init_scripts().await?;
157        Ok(broker)
158      }
159    }
160  }

169  /// 获取 PubSub 连接
170  /// Get PubSub connection
171  pub async fn get_pubsub(&self) -> Result<RedisPubSub> {
172    match &self.client {
173      RedisClient::Single(client) => {
174        let pubsub = client.get_async_pubsub().await?;
175        Ok(RedisPubSub::Single(pubsub))
176      }
177      #[cfg(feature = "cluster")]
178      RedisClient::Cluster {
179        client,
180        push_receiver,
181      } => {
182        // 对于集群模式，从 push_receiver 取出接收器并获取连接
183        // For cluster mode, take receiver from push_receiver and get connection
184        let mut receiver_option = push_receiver.lock().await;
185        if let Some(receiver) = receiver_option.take() {
186          // 获取一个集群连接用于订阅
187          // Get a cluster connection for subscription
188          let connection = client.get_async_connection().await?;
189          let cluster_pubsub = ClusterPubSubConnection::from_receiver(receiver, connection);
190          Ok(RedisPubSub::Cluster(cluster_pubsub))
191        } else {
192          Err(Error::other(
193            "Cluster PubSub receiver has already been taken. You can only create one PubSub connection for cluster mode."
194          ))
195        }
196      }
197      #[cfg(feature = "sentinel")]
198      RedisClient::Sentinel { client, .. } => {
199        // Lock the sentinel client and obtain an async pubsub from the resolved master
200        let mut guard = client.lock().await;
201        let master_client = guard.async_get_client().await?;
202        let pubsub = master_client.get_async_pubsub().await?;
203        Ok(RedisPubSub::Sentinel(pubsub))
204      }
205    }
206  }

224  /// 从字节解码任务消息,这里不需要修复任何编码错误
225  /// Decode task message from bytes, no need to fix any encoding errors here
226  pub fn decode_task_message(&self, data: &[u8]) -> Result<TaskMessage> {
227    match TaskMessage::decode(data) {
228      Ok(msg) => Ok(msg),
229      // 不要处理这里的错误，直接返回错误就可以了
230      // Do not handle errors here, just return the error
231      Err(decode_err) => Err(Error::ProtoDecode(decode_err)),
232    }
233  }

235  /// 从Task创建TaskMessage
236  /// Create TaskMessage from Task
237  pub(crate) fn task_to_message(&self, task: &Task) -> TaskMessage {
238    TaskMessage {
239      r#type: task.task_type.clone(),
240      payload: task.payload.clone(),
241      headers: task.headers.clone(),
242      id: task
243        .options
244        .task_id
245        .clone()
246        .unwrap_or(Uuid::new_v4().to_string()),
247      queue: if task.options.queue.clone().is_empty() {
248        DEFAULT_QUEUE_NAME.to_string()
249      } else {
250        task.options.queue.clone()
251      },
252      retry: task.options.max_retry,
253      retried: 0,
254      error_msg: String::new(),
255      last_failed_at: 0,
256      timeout: task
257        .options
258        .timeout
259        .map(|d| d.as_secs() as i64)
260        .unwrap_or(0),
261      deadline: task.options.deadline.map(|d| d.timestamp()).unwrap_or(0),
262      unique_key: String::new(),
263      group_key: task.options.group.clone().unwrap_or_default(),
264      retention: task
265        .options
266        .retention
267        .map(|d| d.as_secs() as i64)
268        .unwrap_or(0),
269      completed_at: 0,
270    }
271  }

273  /// 批量写入 scheduler entries，兼容 Go 版 asynq
274  /// Batch write scheduler entries, compatible with Go version asynq
275  pub async fn write_scheduler_entries(
276    &self,
277    entries: &[SchedulerEntry],
278    scheduler_id: &str,
279    ttl_secs: u64,
280  ) -> Result<()> {
281    let mut args: Vec<RedisArg> = Vec::new();
282    args.push(RedisArg::Int(ttl_secs as i64));
283    for entry in entries {
284      let mut buf = Vec::new();
285      entry
286        .encode(&mut buf)
287        .map_err(|e| Error::other(format!("prost encode error: {e}")))?;
288      args.push(RedisArg::Bytes(buf));
289    }
290    let key = keys::scheduler_entries_key(scheduler_id);
291    let mut conn = self.get_async_connection().await?;
292    let _: () = self
293      .script_manager
294      .eval_script(
295        &mut conn,
296        "write_scheduler_entries",
297        std::slice::from_ref(&key),
298        &args,
299      )
300      .await?;
301    // 新增：ZADD 到全局 ZSET，score 为过期时间戳
302    // New: ZADD to global ZSET, score is the expiration timestamp
303    let zset_key = keys::ALL_SCHEDULERS;
304    let expire_at = chrono::Utc::now().timestamp() + ttl_secs as i64;
305    let _: () = conn.zadd(zset_key, &key, expire_at).await?;
306
307    Ok(())
308  }

310  /// 记录调度事件，兼容 Go 版 asynq
311  /// Record scheduling event, compatible with Go version asynq
312  pub async fn record_scheduler_enqueue_event(
313    &self,
314    event: &SchedulerEnqueueEvent,
315    entry_id: &str,
316  ) -> Result<()> {
317    let mut buf = Vec::new();
318    event
319      .encode(&mut buf)
320      .map_err(|e| Error::other(format!("prost encode error: {e}")))?;
321    let key = keys::scheduler_history_key(entry_id);
322    let mut conn = self.get_async_connection().await?;
323    let args = vec![
324      RedisArg::Int(event.enqueue_time.map(|x| x.seconds).unwrap_or(0)),
325      RedisArg::Bytes(buf),
326      RedisArg::Int(1000),
327    ];
328    self
329      .script_manager
330      .eval_script::<()>(&mut conn, "record_scheduler_enqueue_event", &[key], &args)
331      .await?;
332    Ok(())
333  }

335  /// 通过脚本获取所有 SchedulerEntry，兼容 Go 版 asynq
336  /// Get all SchedulerEntry through script, compatible with Go version asynq
337  pub async fn scheduler_entries_script(
338    &self,
339    scheduler_id: &str,
340  ) -> Result<std::collections::HashMap<String, Vec<u8>>> {
341    let key = keys::scheduler_entries_key(scheduler_id);
342    let mut conn = self.get_async_connection().await?;
343    // 调用脚本，假设脚本返回 [key1, value1, key2, value2, ...]
344    // Call the script, assuming the script returns [key1, value1, key2, value2, ...]
345    let result: Vec<Vec<u8>> = self
346      .script_manager
347      .eval_script(&mut conn, "get_scheduler_entries", &[key], &[])
348      .await?;
349    let mut map = std::collections::HashMap::new();
350    let mut iter = result.chunks_exact(2);
351    while let Some([k, v]) = iter.next() {
352      let key_str = String::from_utf8_lossy(k).to_string();
353      map.insert(key_str, v.clone());
354    }
355    Ok(map)
356  }

371  /// 删除 scheduler entries 数据，兼容 Go 版 asynq
372  /// Delete scheduler entries data, compatible with Go version asynq
373  pub async fn clear_scheduler_entries(&self, scheduler_id: &str) -> Result<()> {
374    let key = keys::scheduler_entries_key(scheduler_id);
375    let zset_key = keys::ALL_SCHEDULERS;
376    let mut conn = self.get_async_connection().await?;
377    // ZREM 全局调度器 ZSET
378    // ZREM global scheduler ZSET
379    let _: () = conn
380      .zrem(zset_key, &key)
381      .await
382      .map_err(|e| Error::other(format!("redis ZREM error: {e}")))?;
383    // DEL 具体 entry 列表
384    // DEL specific entry list
385    let _: () = conn
386      .del(&key)
387      .await
388      .map_err(|e| Error::other(format!("redis DEL error: {e}")))?;
389    Ok(())
390  }
