97/// Format a duration (milliseconds) as a human-readable string, e.g. "7m29.47s".
98fn format_duration_ms(ms: u64) -> String {
99  let minutes = ms / 60000;
100  let remaining_ms = ms % 60000;
101  let secs = remaining_ms as f64 / 1000.0;
102  if minutes > 0 {
103    format!("{minutes}m{secs:.2}s")
104  } else {
105    format!("{secs:.2}s")
106  }
107}

193/// GET /api/queues – list all queues with stats
194#[utoipa::path(
195  get,
196  path = "/api/queues",
197  responses(
198    (status = 200, description = "List of all queues with statistics", body = QueueListResponse),
199    (status = 503, description = "Inspector not available", body = ApiError),
200  ),
201  tag = "queues",
202)]
203pub async fn list_queues(State(state): State<Arc<AppState>>) -> impl IntoResponse {
204  let Some(ref inspector) = state.inspector else {
205    return (
206      StatusCode::SERVICE_UNAVAILABLE,
207      Json(serde_json::json!({ "error": "Inspector not available" })),
208    );
209  };
210
211  match inspector.get_all_queue_stats().await {
212    Ok(stats) => {
213      // Aggregate stats by queue name (sum counters for duplicate queue entries)
214      let mut agg: HashMap<String, AggCounts> = HashMap::new();
215      // Aggregate using a small named struct for clarity
216      for qs in stats {
217        let e = agg.entry(qs.name).or_default();
218        e.active += qs.active;
219        e.pending += qs.pending;
220        e.scheduled += qs.scheduled;
221        e.retry += qs.retry;
222        e.archived += qs.archived;
223        e.completed += qs.completed;
224        e.aggregating += qs.aggregating;
225      }
226
227      let mut responses: Vec<QueueStatsResponse> = Vec::new();
228      for (
229        queue_name,
230        AggCounts {
231          active,
232          pending,
233          scheduled,
234          retry,
235          archived,
236          completed,
237          aggregating,
238        },
239      ) in agg
240      {
241        let info = inspector.get_queue_info(&queue_name).await.ok();
242        let paused = info.as_ref().map(|i| i.paused).unwrap_or(false);
243        let latency_msec = info
244          .as_ref()
245          .map(|i| i.latency.as_millis() as u64)
246          .unwrap_or(0);
247        let processed = info.as_ref().map(|i| i.processed as i64).unwrap_or(0);
248        let failed = info.as_ref().map(|i| i.failed as i64).unwrap_or(0);
249        responses.push(QueueStatsResponse {
250          queue: queue_name,
251          active,
252          pending,
253          scheduled,
254          retry,
255          archived,
256          completed,
257          aggregating,
258          processed,
259          // succeeded = tasks that completed without failure (processed includes both success and failure)
260          succeeded: (processed - failed).max(0),
261          failed,
262          paused,
263          size: info.as_ref().map(|i| i.size as i64).unwrap_or(0),
264          groups: info.as_ref().map(|i| i.groups as i64).unwrap_or(0),
265          latency_msec,
266          display_latency: format_duration_ms(latency_msec),
267          memory_usage_bytes: info.as_ref().map(|i| i.memory_usage).unwrap_or(0),
268          timestamp: info
269            .as_ref()
270            .map(|i| i.timestamp.to_rfc3339())
271            .unwrap_or_default(),
272        });
273      }
274
275      let body = QueueListResponse { queues: responses };
276      (StatusCode::OK, Json(serde_json::to_value(body).unwrap()))
277    }
278    Err(e) => (
279      StatusCode::INTERNAL_SERVER_ERROR,
280      Json(serde_json::json!({ "error": e.to_string() })),
281    ),
282  }
283}

345/// GET /api/queues/:queue/tasks – list tasks in a queue filtered by state
346#[utoipa::path(
347  get,
348  path = "/api/queues/{queue}/tasks",
349  params(
350    ("queue" = String, Path, description = "Queue name"),
351    TaskListQuery,
352  ),
353  responses(
354    (status = 200, description = "Paginated list of tasks", body = TaskListResponse),
355    (status = 503, description = "Inspector not available", body = ApiError),
356  ),
357  tag = "tasks",
358)]
359pub async fn list_tasks(
360  State(state): State<Arc<AppState>>,
361  Path(queue): Path<String>,
362  Query(params): Query<TaskListQuery>,
363) -> impl IntoResponse {
364  let Some(ref inspector) = state.inspector else {
365    return (
366      StatusCode::SERVICE_UNAVAILABLE,
367      Json(serde_json::json!({ "error": "Inspector not available" })),
368    );
369  };
370
371  let task_state = match params.state.as_str() {
372    "active" => TaskState::Active,
373    "pending" => TaskState::Pending,
374    "scheduled" => TaskState::Scheduled,
375    "retry" => TaskState::Retry,
376    "archived" => TaskState::Archived,
377    "completed" => TaskState::Completed,
378    "aggregating" => TaskState::Aggregating,
379    _ => TaskState::Pending,
380  };
381
382  let pagination = Pagination {
383    size: params.size.clamp(1, MAX_PAGE_SIZE),
384    page: params.page.max(0),
385  };
386
387  match inspector.list_tasks(&queue, task_state, pagination).await {
388    Ok(tasks) => {
389      let responses: Vec<TaskInfoResponse> = tasks
390        .into_iter()
391        .map(|t| TaskInfoResponse {
392          id: t.id,
393          queue: t.queue,
394          task_type: t.task_type,
395          state: t.state.as_str().to_string(),
396          max_retry: t.max_retry,
397          retried: t.retried,
398          last_err: t.last_err,
399          last_failed_at: t.last_failed_at.map(|dt| dt.to_rfc3339()),
400          next_process_at: t.next_process_at.map(|dt| dt.to_rfc3339()),
401          completed_at: t.completed_at.map(|dt| dt.to_rfc3339()),
402          deadline: t.deadline.map(|dt| dt.to_rfc3339()),
403          group: t.group,
404          is_orphaned: t.is_orphaned,
405        })
406        .collect();
407
408      let resp = TaskListResponse {
409        tasks: responses,
410        state: params.state.clone(),
411        queue,
412        page: params.page,
413        size: params.size,
414      };
415      (StatusCode::OK, Json(serde_json::to_value(resp).unwrap()))
416    }
417    Err(e) => (
418      StatusCode::INTERNAL_SERVER_ERROR,
419      Json(serde_json::json!({ "error": e.to_string() })),
420    ),
421  }
422}

696    "archived" => inspector.requeue_all_archived_tasks(&queue).await,

698    "scheduled" => inspector.requeue_all_scheduled_tasks(&queue).await,
699    _ => {
700      return (
701        StatusCode::BAD_REQUEST,
702        Json(serde_json::json!({ "error": "Invalid state. Use: archived, retry, scheduled" })),
703      );
704    }
705  };
706
707  match result {
708    Ok(count) => (
709      StatusCode::OK,
710      Json(serde_json::json!({ "message": format!("Requeued {} tasks", count), "count": count })),
711    ),
712    Err(e) => (
713      StatusCode::INTERNAL_SERVER_ERROR,
714      Json(serde_json::json!({ "error": e.to_string() })),
715    ),
716  }
717}
