23/// Handler for processing client messages
24pub struct MessageHandler {
25  broker: Arc<dyn Broker>,
26}

28impl MessageHandler {
29  /// Create a new message handler
30  pub fn new(broker: Arc<dyn Broker>) -> Self {
31    Self { broker }
32  }
33
34  /// Get a reference to the broker
35  pub fn broker(&self) -> &Arc<dyn Broker> {
36    &self.broker
37  }

39  /// Apply tenant prefix to a queue name if tenant is configured
40  fn apply_tenant_prefix(queue: &str) -> String {
41    if let Some(tenant) = Self::get_current_tenant() {
42      tenant.get_tenant_queue(queue)
43    } else {
44      queue.to_string()
45    }
46  }
47
48  /// Get current tenant from task-local variable
49  fn get_current_tenant() -> Option<TenantConfig> {
50    CURRENT_TENANT.try_with(|t| t.clone()).ok().flatten()
51  }

53  /// Handle a client message and return a server response
54  ///
55  /// The tenant configuration is retrieved from the CURRENT_TENANT task-local variable
56  /// which is set by the authentication middleware. Queue names will be prefixed with
57  /// the tenant ID if a tenant is configured.
58  pub async fn handle(&self, msg: ClientMessage) -> Result<ServerMessage> {
59    match msg {
60      ClientMessage::Enqueue(req) => self.handle_enqueue(req).await,
61      ClientMessage::EnqueueUnique(req) => self.handle_enqueue_unique(req).await,
62      ClientMessage::Schedule(req) => self.handle_schedule(req).await,
63      ClientMessage::ScheduleUnique(req) => self.handle_schedule_unique(req).await,
64      ClientMessage::Dequeue(req) => self.handle_dequeue(req).await,
65      ClientMessage::Done(req) => self.handle_done(req).await,
66      ClientMessage::MarkComplete(req) => self.handle_mark_complete(req).await,
67      ClientMessage::Retry(req) => self.handle_retry(req).await,
68      ClientMessage::Archive(req) => self.handle_archive(req).await,
69      ClientMessage::Ping => Ok(ServerMessage::Pong),
70      ClientMessage::SubscribeCancellation => {
71        // Subscription is handled at the WebSocket level
72        Ok(ServerMessage::Success)
73      }
74      ClientMessage::PublishCancellation { task_id } => {
75        self.handle_publish_cancellation(&task_id).await
76      }
77      ClientMessage::AddToGroup(req) => self.handle_add_to_group(req).await,
78      ClientMessage::AddToGroupUnique(req) => self.handle_add_to_group_unique(req).await,
79      ClientMessage::ListGroups(req) => self.handle_list_groups(req).await,
80      ClientMessage::AggregationCheck(req) => self.handle_aggregation_check(req).await,
81      ClientMessage::ReadAggregationSet(req) => self.handle_read_aggregation_set(req).await,
82      ClientMessage::DeleteAggregationSet(req) => self.handle_delete_aggregation_set(req).await,
83    }
84  }

86  /// Handle enqueue request
87  async fn handle_enqueue(&self, req: EnqueueRequest) -> Result<ServerMessage> {
88    let task = self.create_task_from_request(&req)?;
89    let info = self.broker.enqueue(&task).await?;
90    Ok(ServerMessage::TaskInfo(TaskInfoResponse {
91      id: info.id,
92      queue: info.queue,
93      task_type: info.task_type,
94      state: info.state.as_str().to_string(),
95    }))
96  }

98  /// Handle enqueue unique request
99  async fn handle_enqueue_unique(&self, req: EnqueueUniqueRequest) -> Result<ServerMessage> {
100    let task = self.create_task_from_request(&req.enqueue)?;
101    let ttl = Duration::from_secs(req.ttl_seconds);
102    let info = self.broker.enqueue_unique(&task, ttl).await?;
103    Ok(ServerMessage::TaskInfo(TaskInfoResponse {
104      id: info.id,
105      queue: info.queue,
106      task_type: info.task_type,
107      state: info.state.as_str().to_string(),
108    }))
109  }

111  /// Handle schedule request
112  async fn handle_schedule(&self, req: ScheduleRequest) -> Result<ServerMessage> {
113    let task = self.create_task_from_request(&req.enqueue)?;
114    let process_at = DateTime::from_timestamp(req.process_at, 0).unwrap_or_else(Utc::now);
115    let info = self.broker.schedule(&task, process_at).await?;
116    Ok(ServerMessage::TaskInfo(TaskInfoResponse {
117      id: info.id,
118      queue: info.queue,
119      task_type: info.task_type,
120      state: info.state.as_str().to_string(),
121    }))
122  }

124  /// Handle schedule unique request
125  async fn handle_schedule_unique(&self, req: ScheduleUniqueRequest) -> Result<ServerMessage> {
126    let task = self.create_task_from_request(&req.schedule.enqueue)?;
127    let process_at = DateTime::from_timestamp(req.schedule.process_at, 0).unwrap_or_else(Utc::now);
128    let ttl = Duration::from_secs(req.ttl_seconds);
129    let info = self.broker.schedule_unique(&task, process_at, ttl).await?;
130    Ok(ServerMessage::TaskInfo(TaskInfoResponse {
131      id: info.id,
132      queue: info.queue,
133      task_type: info.task_type,
134      state: info.state.as_str().to_string(),
135    }))
136  }

138  /// Handle dequeue request
139  async fn handle_dequeue(&self, req: DequeueRequest) -> Result<ServerMessage> {
140    // Apply tenant prefix to all queues
141    let queues: Vec<String> = req
142      .queues
143      .iter()
144      .map(|q| Self::apply_tenant_prefix(q))
145      .collect();
146    let result = self.broker.dequeue(&queues).await?;
147    let response = result.map(|msg| self.task_message_to_response(&msg));
148    Ok(ServerMessage::DequeueResult(response))
149  }
150
151  /// Handle task done request
152  async fn handle_done(&self, req: TaskDoneRequest) -> Result<ServerMessage> {
153    let msg = self.create_task_message_from_request(&req)?;
154    self.broker.done(&msg).await?;
155    Ok(ServerMessage::Success)
156  }
157
158  /// Handle mark complete request
159  async fn handle_mark_complete(&self, req: TaskDoneRequest) -> Result<ServerMessage> {
160    let msg = self.create_task_message_from_request(&req)?;
161    self.broker.mark_as_complete(&msg).await?;
162    Ok(ServerMessage::Success)
163  }
164
165  /// Handle retry request
166  async fn handle_retry(&self, req: RetryRequest) -> Result<ServerMessage> {
167    let msg = self.create_task_message_from_request(&req.task)?;
168    let process_at = DateTime::from_timestamp(req.process_at, 0).unwrap_or_else(Utc::now);
169    self
170      .broker
171      .retry(&msg, process_at, &req.error_msg, req.is_failure)
172      .await?;
173    Ok(ServerMessage::Success)
174  }

176  /// Handle archive request
177  async fn handle_archive(&self, req: ArchiveRequest) -> Result<ServerMessage> {
178    let msg = self.create_task_message_from_request(&req.task)?;
179    self.broker.archive(&msg, &req.error_msg).await?;
180    Ok(ServerMessage::Success)
181  }

189  /// Create a Task from an enqueue request
190  fn create_task_from_request(&self, req: &EnqueueRequest) -> Result<Task> {
191    let payload = BASE64_STANDARD
192      .decode(&req.payload)
193      .map_err(|e| Error::invalid_message(format!("Invalid base64 payload: {}", e)))?;
194
195    let mut task = Task::new(&req.task_type, &payload)
196      .map_err(|e| Error::invalid_message(format!("Invalid task: {}", e)))?;
197
198    // Apply tenant prefix to queue name
199    if let Some(ref queue) = req.queue {
200      let tenant_queue = Self::apply_tenant_prefix(queue);
201      task = task.with_queue(&tenant_queue);
202    } else if let Some(tenant) = Self::get_current_tenant() {
203      // If no queue specified, use tenant-prefixed default queue
204      task = task.with_queue(tenant.get_tenant_queue("default"));
205    }
206
207    if let Some(max_retry) = req.max_retry {
208      task = task.with_max_retry(max_retry);
209    }
210    if let Some(ref task_id) = req.task_id {
211      task = task.with_task_id(task_id);
212    }
213
214    // Add headers
215    for (key, value) in &req.headers {
216      task.headers.insert(key.clone(), value.clone());
217    }
218
219    Ok(task)
220  }

222  /// Create a TaskMessage from a done request
223  fn create_task_message_from_request(&self, req: &TaskDoneRequest) -> Result<TaskMessage> {
224    let payload = BASE64_STANDARD
225      .decode(&req.payload)
226      .map_err(|e| Error::invalid_message(format!("Invalid base64 payload: {}", e)))?;
227
228    // Apply tenant prefix to queue name
229    let queue = Self::apply_tenant_prefix(&req.queue);
230
231    Ok(TaskMessage {
232      id: req.task_id.clone(),
233      queue,
234      r#type: req.task_type.clone(),
235      payload,
236      ..Default::default()
237    })
238  }
239
240  /// Convert a TaskMessage to a response
241  fn task_message_to_response(&self, msg: &TaskMessage) -> TaskMessageResponse {
242    TaskMessageResponse {
243      id: msg.id.clone(),
244      queue: msg.queue.clone(),
245      task_type: msg.r#type.clone(),
246      payload: BASE64_STANDARD.encode(&msg.payload),
247      headers: msg.headers.clone(),
248      retry: msg.retry,
249      retried: msg.retried,
250      error_msg: msg.error_msg.clone(),
251      timeout: msg.timeout,
252      deadline: msg.deadline,
253      group_key: msg.group_key.clone(),
254    }
255  }

295  /// Handle aggregation check request
296  async fn handle_aggregation_check(&self, req: AggregationCheckRequest) -> Result<ServerMessage> {
297    let queue = Self::apply_tenant_prefix(&req.queue);
298    let aggregation_delay = Duration::from_secs(req.aggregation_delay_seconds);
299    let max_delay = Duration::from_secs(req.max_delay_seconds);
300    let set_id = self
301      .broker
302      .aggregation_check(
303        &queue,
304        &req.group,
305        aggregation_delay,
306        max_delay,
307        req.max_size,
308      )
309      .await?;
310    Ok(ServerMessage::AggregationSetId(set_id))
311  }

313  /// Handle read aggregation set request
314  async fn handle_read_aggregation_set(
315    &self,
316    req: ReadAggregationSetRequest,
317  ) -> Result<ServerMessage> {
318    let queue = Self::apply_tenant_prefix(&req.queue);
319    let tasks = self
320      .broker
321      .read_aggregation_set(&queue, &req.group, &req.set_id)
322      .await?;
323    let responses: Vec<TaskMessageResponse> = tasks
324      .iter()
325      .map(|msg| self.task_message_to_response(msg))
326      .collect();
327    Ok(ServerMessage::AggregationSet(responses))
328  }
